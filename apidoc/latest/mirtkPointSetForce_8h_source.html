38 /**
39  * Base class for point set force terms
40  *
41  * Subclasses implement in particular internal and external forces for
42  * deformable surface models. Internal force terms may further be used to
43  * regularize the deformation of a surface during image/point set registration.
44  */
45 class PointSetForce : public EnergyTerm
46 {
47  mirtkAbstractMacro(PointSetForce);
48 
49  // ---------------------------------------------------------------------------
50  // Types
51 public:
52 
53  /// Type of gradient w.r.t a single transformed data point

56  // ---------------------------------------------------------------------------

58 protected:
59 
60  /// Transformed point set
61  mirtkPublicAggregateMacro(RegisteredPointSet, PointSet);
62 
63  /// Number of gradient averaging iterations
64  mirtkPublicAttributeMacro(int, GradientAveraging);
65 
66  /// Whether to only average gradient vectors pointing in the same direction
67  /// as the unsmoothed gradient at the central node (i.e., positive dot product)
68  mirtkPublicAttributeMacro(bool, AverageSignedGradients);
69 
70  /// Whether to only average the magnitude of the gradient vectors
71  mirtkPublicAttributeMacro(bool, AverageGradientMagnitude);
72 
73  /// Whether this force is only acting on the point set surface
74  ///
75  /// This read-only attribute must be set by the subclass constructor.
76  /// It is in particular set by the SurfaceForce and SurfaceConstraint
77  /// constructors.
78  mirtkReadOnlyAttributeMacro(bool, SurfaceForce);
79

82 
83  /// Negative node forces/gradient of external force term

86  /// Size of gradient vector

88 
89  /// Number of summands in gradient computation
90  ///
91  /// Intended for use by EvaluateGradient implementations only.

94 
95  /// Size of count vector

97

99  mirtkAttributeMacro(bool, InitialUpdate);

105  typedef UnorderedMap<string, string> NameMap;

107  typedef NameMap::const_iterator NameMapConstIterator;
108 
109  /// Maps internal point data name to actual unique point data array name
110  NameMap _PointDataName;

112 protected:
113 
114  /// Add given array to point data attributes of deformed point set
115  ///
116  /// This function should be used by subclasses to add point data arrays to
117  /// the point set (or its surface, respectively, if \c _SurfaceForce is \c true).
118  /// The added point data is interpolated at new node positions whenever the
119  /// deformed point set is being remeshed during the optimization.
120  ///
121  /// \param[in] name Name of array. The actual name of the point data array
122  /// will be made unique by this function which stores an
123  /// internal map from the given name to the unique array name.
124  /// \param[in] data Point data array.
125  /// \param[in] global Whether point data array is shared among point forces.
126  /// If \c true, the array \p name is used unmodified such
127  /// that other force terms can reuse the array.
128  void AddPointData(const char *name, vtkSmartPointer<vtkDataArray> &data, bool global = false);
129 
130  /// Add new point data array of given type with specified number of components
131  ///
132  /// This function should be used by subclasses to add point data arrays to
133  /// the point set (or its surface, respectively, if \c _SurfaceForce is \c true).
134  /// The added point data is interpolated at new node positions whenever the
135  /// deformed point set is being remeshed during the optimization.
136  ///
137  /// If an array with the given \p name already exists, it is reused to avoid
138  /// unnecessary allocations unless the data type or number of components mismatch.
139  /// If _NumberOfPoints is set before by PointForce::Initialize, the
140  /// corresponding number of array tuples are allocated by this function.
141  /// Otherwise, the array is only instantiated, but not allocated.
142  ///
143  /// \param[in] name Name of array. The actual name of the point data array
144  /// will be made unique by this function which stores an
145  /// internal map from the given name to the unique array name.
146  /// \param[in] c Number of components.
147  /// \param[in] type Type of data array (e.g., VTK_FLOAT, the default).
148  /// \param[in] global Whether point data array is shared among point forces.
149  /// If \c true, the array \p name is used unmodified such
150  /// that other force terms can reuse the array.
151  ///
152  /// \return Pointer to (newly instantiated) array.
153  vtkDataArray *AddPointData(const char *name, int c = 1, int type = VTK_FLOAT, bool global = false);
154 
155  /// Remove named array from point data attributes of deformed point set
156  ///
157  /// \param[in] name Name of array as used when the array was added before.
158  /// This name may differ from the actual unique array name.
159  void RemovePointData(const char *name);
160 
161  /// Get point data array of deformed point set
162  ///
163  /// \param[in] name Name of array as used when the array was added before.
164  /// This name may differ from the actual unique array name.
165  /// \param[in] optional Whether the array may not exist. If \c false, this
166  /// function raises an error if the array does not exist.
167  ///
168  /// \return Point data array or NULL if not found (only if \p optional = \c true).
169  vtkDataArray *GetPointData(const char *name, bool optional = false) const;
170

173 protected:
174 
175  /// Constructor

177 
178  /// Copy constructor
179  PointSetForce(const PointSetForce &);
180 
181  /// Assignment operator
182  PointSetForce &operator =(const PointSetForce &);
183 
184  /// Allocate memory for (non-parametric) gradient
185  void AllocateGradient(int);
186 
187  /// Allocate _Count memory

203  void Init();

206  vtkSmartPointer<vtkPoints> GetInitialPoints() const;

208 public:
209 
210  /// Initialize force term once input and parameters have been set
211  virtual void Initialize();
212 
213  /// Reinitialize force term after change of input topology
214  ///
215  /// This function is called in particular when an input surface has been
216  /// reparameterized, e.g., by a local remeshing filter.
217  virtual void Reinitialize();
218 
219  // ---------------------------------------------------------------------------
220  // Evaluation
221 
222  /// Update moving input points and internal state of force term
223  virtual void Update(bool = true);
224 
225 protected:
226 
227  /// Evaluate gradient of force term
228  ///
229  /// \param[in,out] gradient Gradient to which the computed gradient of the
230  /// force term is added after multiplying by \p weight.
231  /// \param[in] step Step length for finite differences (unused).
232  /// \param[in] weight Weight of force term.
233  virtual void EvaluateGradient(double *gradient, double step, double weight);
234 
235  // ---------------------------------------------------------------------------
236  // Debugging
237 public:
238 
239  /// Write input of force term
240  virtual void WriteDataSets(const char *, const char *, bool = true) const;
241 
242  /// Write gradient of force term
243  virtual void WriteGradient(const char *, const char *) const;
