33 /**
34  * N-dimensional polynomial regression model.
35  *
36  * This implementation is based on the polyfitn and polyvaln MATLAB functions
37  * written and distributed on MathCentral by John D'Errico.
38  *
39  * \sa http://uk.mathworks.com/matlabcentral/fileexchange/34765-polyfitn
40  */
41 class PolynomialModel : public Object
42 {
43  mirtkObjectMacro(PolynomialModel);
44 
45  // ---------------------------------------------------------------------------
46  // Attributes
47 
48  /// Order of the polynomial model, i.e., the highest exponent
49  mirtkReadOnlyAttributeMacro(int, Order);
50 
51  /// Dimension of the independent variable space
52  mirtkReadOnlyAttributeMacro(int, Dimension);
53 
54  /// Exponents of the model for each independent variable
55  ///
56  /// Each column corresponds to one variable, and each
57  /// row to the exponents of each term for one variable.
58  mirtkReadOnlyAttributeMacro(Matrix, ModelTerms);
59 
60  /// Coefficients of each model term
61  mirtkReadOnlyAttributeMacro(Vector, Coefficients);
62 
63  /// Copy attributes of this class from another instance
64  void CopyAttributes(const PolynomialModel &);
65 
66  // ---------------------------------------------------------------------------
67  // Construction/Destruction
68 public:
69 
70  /// Constructor
71  ///
72  /// \param[in] order Order of the polynomial, i.e., highest model term exponent.
73  PolynomialModel(int order = 2);
74 
75  /// Copy constructor
76  PolynomialModel(const PolynomialModel &);
77 
78  /// Assignment operator
79  PolynomialModel &operator =(const PolynomialModel &);
80 
81  /// Destructor
82  virtual ~PolynomialModel();
83 
84 protected:
85 
86  /// Build complete polynomial regression model of given order
87  ///
88  /// \param[in] p Dimension of independent variable space.
89  /// \param[in] order Order of polynomial model, i.e., highest model term exponent.
90  /// When non-positive, the current order of the model is used.
91  ///
92  /// \returns Number of model terms.
93  int Initialize(int p, int order = 0);
94 
95  // ---------------------------------------------------------------------------
96  // Regression
97 public:
98 
99  /// Fits polynomial regression model to one or more independent variables
100  ///
101  /// \param[in] x Values of independent variables, a matrix of size (n x p), where
102  /// n is the number of data points and
103  /// p is the dimension of the independent variable space.
104  /// \param[in] y Values of dependent variable.
105  /// \param[in] order Order of polynomial model, i.e., highest model term exponent.
106  /// When non-positive, the current order of the model is used.
107  ///
108  /// \returns RMS error of the regression.
109  double Fit(const Matrix &x, const Vector &y, int order = 0);
110 
111  /// Fits polynomial regression model to one or more independent variables
112  ///
113  /// \param[in] x Values of independent variables.
114  /// The dimension of the independent variable space is 3.
115  /// \param[in] y Values of dependent variable.
116  /// \param[in] order Order of polynomial model, i.e., highest model term exponent.
117  /// When non-positive, the current order of the model is used.
118  ///
119  /// \returns RMS error of the regression.
120  double Fit(const PointSet &x, const Vector &y, int order = 0);
121 
122  /// Fits polynomial regression model to one independent variable
123  ///
124  /// \param[in] x Values of independent variable.
125  /// \param[in] y Values of dependent variable.
126  /// \param[in] order Order of polynomial model, i.e., highest model term exponent.
127  /// When non-positive, the current order of the model is used.
128  ///
129  /// \returns RMS error of the regression.
130  double Fit(const Vector &x, const Vector &y, int order = 0);
131 
132  // ---------------------------------------------------------------------------
133  // Evaluation
134 
135  /// Evaluates polynomial for n p-dimensional data points
136  ///
137  /// \param[in] x Matrix of size (n x p), where n is the number of data points
138  /// to evaluate, and p is the dimension of the independent variable
139  /// space of the polynomial model.
140  ///
141  /// \returns Regressed value.
142  Vector Evaluate(const Matrix &x) const;
143 
144  /// Evaluates polynomial for one p-dimensional data point
145  ///
146  /// \param[in] x Data point with p values, where p is the dimension of the
147  /// independent variable space of the polynomial model.
148  ///
149  /// \returns Regressed value.
150  double Evaluate(const Vector &x) const;
151 
152  /// Evaluates polynomial for one 3D point
153  ///
154  /// \param[in] x Data point. The dimension of the model must be 3.
155  ///
156  /// \returns Regressed value.
157  double Evaluate(const Point &x) const;
158 
159 };
