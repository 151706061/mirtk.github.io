55 /// Case insensitive string comparison
56 inline bool iequal(char const *a, char const *b)
57 {
58  while (*a && *b) {
59  if (tolower(*a) != tolower(*b)) return false;
60  ++a, ++b;
61  }
62  return false;
63 }

112 /// Convert string to boolean value
113 template <>
114 inline bool FromString(const char *str, bool &value)
115 {
116  if (strcmp(str, "yes") == 0 || strcmp(str, "Yes") == 0 || strcmp(str, "YES") == 0) {
117  value = true;
118  return true;
119  } else if (strcmp(str, "no") == 0 || strcmp(str, "No") == 0 || strcmp(str, "NO") == 0) {
120  value = false;
121  return true;
122  } else if (strcmp(str, "true") == 0 || strcmp(str, "True") == 0 || strcmp(str, "TRUE") == 0) {
123  value = true;
124  return true;
125  } else if (strcmp(str, "false") == 0 || strcmp(str, "False") == 0 || strcmp(str, "FALSE") == 0) {
126  value = false;
127  return true;
128  } else if (strcmp(str, "on") == 0 || strcmp(str, "On") == 0 || strcmp(str, "ON") == 0) {
129  value = true;
130  return true;
131  } else if (strcmp(str, "off") == 0 || strcmp(str, "Off") == 0 || strcmp(str, "OFF") == 0) {
132  value = false;
133  return true;
134  } else {
135  std::istringstream is(str);
136  return !(is >> value).fail() && is.eof();
137  }
138 }

140 /// Convert string to float value
141 template <>
142 inline bool FromString(const char *str, float &value)
143 {
144  if (strcmp(str, "nan") == 0 || strcmp(str, "NaN") == 0) {
145  value = std::numeric_limits<float>::quiet_NaN();
146  return true;
147  } else if (strcmp(str, "-inf") == 0 || strcmp(str, "-Inf") == 0) {
148  value = -std::numeric_limits<float>::infinity();
149  return true;
150  } else if (strcmp(str, "+inf") == 0 || strcmp(str, "inf") == 0 || strcmp(str, "+Inf") == 0 || strcmp(str, "Inf") == 0) {
151  value = +std::numeric_limits<float>::infinity();
152  return true;
153  } else {
154  std::istringstream is(str);
155  return !(is >> value).fail() && is.eof();
156  }
157 }

178 /// Convert numeric value to string
179 template <typename T>
180 string ToString(const T &value, int w = 0, char c = ' ', bool left = false)
181 {
182  std::ostringstream os;
183  os.fill(c);
184  if (left) os << std::left;
185  else os << std::right;
186  os << std::setw(w) << value;
187  return os.str();
188 }

190 /// Convert boolean value to string

192 inline string ToString(const bool &value, int w, char c, bool left)

196  if (left) os << std::left;
197  else os << std::right;

199  return os.str();
