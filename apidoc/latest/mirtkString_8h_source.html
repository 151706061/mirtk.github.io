63  if (str == NULL || str[0] == '\0') return false;

75 /// Convert string to boolean value
76 template <>
77 inline bool FromString(const char *str, bool &value)
78 {
79  if (strcmp(str, "yes") == 0 || strcmp(str, "Yes") == 0 || strcmp(str, "YES") == 0) {
80  value = true;
81  return true;
82  } else if (strcmp(str, "no") == 0 || strcmp(str, "No") == 0 || strcmp(str, "NO") == 0) {
83  value = false;
84  return true;
85  } else if (strcmp(str, "true") == 0 || strcmp(str, "True") == 0 || strcmp(str, "TRUE") == 0) {
86  value = true;
87  return true;
88  } else if (strcmp(str, "false") == 0 || strcmp(str, "False") == 0 || strcmp(str, "FALSE") == 0) {
89  value = false;
90  return true;
91  } else if (strcmp(str, "on") == 0 || strcmp(str, "On") == 0 || strcmp(str, "ON") == 0) {
92  value = true;
93  return true;
94  } else if (strcmp(str, "off") == 0 || strcmp(str, "Off") == 0 || strcmp(str, "OFF") == 0) {
95  value = false;
96  return true;
97  } else {
98  std::istringstream is(str);
99  return !(is >> value).fail() && is.eof();
100  }

103 /// Convert string to float value
104 template <>
105 inline bool FromString(const char *str, float &value)
106 {
107  if (strcmp(str, "nan") == 0 || strcmp(str, "NaN") == 0) {
108  value = std::numeric_limits<float>::quiet_NaN();
109  return true;
110  } else if (strcmp(str, "-inf") == 0 || strcmp(str, "-Inf") == 0) {
111  value = -std::numeric_limits<float>::infinity();
112  return true;
113  } else if (strcmp(str, "+inf") == 0 || strcmp(str, "inf") == 0 || strcmp(str, "+Inf") == 0 || strcmp(str, "Inf") == 0) {
114  value = +std::numeric_limits<float>::infinity();
115  return true;
116  } else {
117  std::istringstream is(str);
118  return !(is >> value).fail() && is.eof();
119  }
120 }

141 /// Convert numeric value to string
142 template <typename T>
143 string ToString(const T &value, int w = 0, char c = ' ', bool left = false)
144 {
145  std::ostringstream os;
146  os.fill(c);
147  if (left) os << std::left;
148  else os << std::right;
149  os << std::setw(w) << value;
150  return os.str();
151 }

153 /// Convert boolean value to string

155 inline string ToString(const bool &value, int w, char c, bool left)

162  return os.str();

165 /// Write "<name> = <value>" configuration entry to output stream
166 template <class TValue>
167 inline void PrintParameter(std::ostream &os, const char *name, const TValue &value)
168 {
169  const std::streamsize w = os.width(40);
170  os << std::left << name << std::setw(0) << " = " << value << std::endl;
171  os.width(w);
172 }
