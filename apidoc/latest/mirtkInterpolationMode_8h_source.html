57 inline string ToString(const InterpolationMode &m)
58 {
59  switch(m) {
60  case Interpolation_NN: return "NN";
61  case Interpolation_Linear: return "Linear";
62  case Interpolation_FastLinear: return "Fast linear";
63  case Interpolation_BSpline: return "BSpline";
64  case Interpolation_CSpline: return "CSpline";
65  case Interpolation_CubicBSpline: return "Cubic BSpline";
66  case Interpolation_FastCubicBSpline: return "Fast cubic BSpline";
67  case Interpolation_SBased: return "SBased";
68  case Interpolation_Sinc: return "Sinc";
69  case Interpolation_Gaussian: return "Gaussian";
70  case Interpolation_NNWithPadding: return "NN with padding";
71  case Interpolation_LinearWithPadding: return "Linear with padding";
72  case Interpolation_FastLinearWithPadding: return "Fast linear with padding";
73  case Interpolation_BSplineWithPadding: return "BSpline with padding";
74  case Interpolation_CubicBSplineWithPadding: return "Cubic BSpline with padding";

77  case Interpolation_SBasedWithPadding: return "SBased with padding";

85 inline bool FromString(const char *str, InterpolationMode &m)
86 {
87  if (strcmp(str, "NN") == 0) m = Interpolation_NN;
88  else if (strcmp(str, "Linear") == 0) m = Interpolation_Linear;
89  else if (strcmp(str, "Fast linear") == 0) m = Interpolation_FastLinear;
90  else if (strcmp(str, "BSpline") == 0) m = Interpolation_BSpline;
91  else if (strcmp(str, "CSpline") == 0) m = Interpolation_CSpline;
92  else if (strcmp(str, "Cubic BSpline") == 0) m = Interpolation_CubicBSpline;
93  else if (strcmp(str, "Fast cubic BSpline") == 0) m = Interpolation_FastCubicBSpline;
94  else if (strcmp(str, "SBased") == 0) m = Interpolation_SBased;
95  else if (strcmp(str, "Sinc") == 0) m = Interpolation_Sinc;
96  else if (strcmp(str, "Gaussian") == 0) m = Interpolation_Gaussian;
97  else if (strcmp(str, "NN with padding") == 0) m = Interpolation_NNWithPadding;
98  else if (strcmp(str, "NN (with padding)") == 0) m = Interpolation_NNWithPadding;
99  else if (strcmp(str, "Linear with padding") == 0) m = Interpolation_LinearWithPadding;
100  else if (strcmp(str, "Linear (with padding)") == 0) m = Interpolation_LinearWithPadding;
101  else if (strcmp(str, "Fast linear with padding") == 0) m = Interpolation_FastLinearWithPadding;
102  else if (strcmp(str, "Fast linear (with padding)") == 0) m = Interpolation_FastLinearWithPadding;
103  else if (strcmp(str, "BSpline with padding") == 0) m = Interpolation_BSplineWithPadding;
104  else if (strcmp(str, "BSpline (with padding)") == 0) m = Interpolation_BSplineWithPadding;
105  else if (strcmp(str, "Cubic BSpline with padding") == 0) m = Interpolation_CubicBSplineWithPadding;
106  else if (strcmp(str, "Cubic BSpline (with padding)") == 0) m = Interpolation_CubicBSplineWithPadding;
107  else if (strcmp(str, "Fast cubic BSpline with padding") == 0) m = Interpolation_FastCubicBSplineWithPadding;
108  else if (strcmp(str, "Fast cubic BSpline (with padding)") == 0) m = Interpolation_FastCubicBSplineWithPadding;
109  else if (strcmp(str, "CSpline with padding") == 0) m = Interpolation_CSplineWithPadding;
110  else if (strcmp(str, "CSpline (with padding)") == 0) m = Interpolation_CSplineWithPadding;
111  else if (strcmp(str, "SBased with padding") == 0) m = Interpolation_SBasedWithPadding;
112  else if (strcmp(str, "SBased (with padding)") == 0) m = Interpolation_SBasedWithPadding;
113  else if (strcmp(str, "Sinc with padding") == 0) m = Interpolation_SincWithPadding;
114  else if (strcmp(str, "Sinc (with padding)") == 0) m = Interpolation_SincWithPadding;
115  else if (strcmp(str, "Gaussian with padding") == 0) m = Interpolation_GaussianWithPadding;
116  else if (strcmp(str, "Gaussian (with padding)") == 0) m = Interpolation_GaussianWithPadding;
117  else return false;
118  return true;
119 }
