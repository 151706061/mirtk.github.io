30 /// Enumeration of implemented numerical integration methods
31 enum FFDIntegrationMethod
32 {
33  FFDIM_Unknown = 0, // keep at begin
34  FFDIM_RKE1, FFDIM_RKE2, FFDIM_RKH2, FFDIM_RK4, // explicit RK
35  FFDIM_RKEH12, FFDIM_RKBS23, FFDIM_RKF45, FFDIM_RKDP45, FFDIM_RKCK45, // embedded RK
36  FFDIM_SS, FFDIM_FastSS, // scaling and squaring
37  FFDIM_NUM // keep at end
38 };

41 /// Convert FFD integration method enumeration value to string
42 inline const char *ToString(FFDIntegrationMethod m)
43 {
44  switch (m) {
45  case FFDIM_SS: return "SS";
46  case FFDIM_FastSS: return "FastSS";
47  case FFDIM_RKE1: return "RKE1";
48  case FFDIM_RKE2: return "RKE2";
49  case FFDIM_RKH2: return "RKH2";
50  case FFDIM_RK4: return "RK4";
51  case FFDIM_RKEH12: return "RKEH12";
52  case FFDIM_RKBS23: return "RKBS23";
53  case FFDIM_RKF45: return "RKF45";
54  case FFDIM_RKDP45: return "RKDP45";
55  case FFDIM_RKCK45: return "RKCK45";
56  default: return "Unknown";
57  }
58 }

61 /// Convert FFD integration method string to enumeration value
62 inline bool FromString(const char *str, FFDIntegrationMethod &m)
63 {
64  // Convert string to all uppercase
65  char *STR = new char[strlen(str) + 3]; // extra space for alias replacement
66  strcpy(STR, str);
67  for (char *p = STR; *p != '\0'; p++) *p = toupper(*p);
68  // Translate aliases of methods to actual identifiers
69  if (strcmp(STR, "SCALINGANDSQUARING") == 0) strcpy(STR, "SS");
70  else if (strcmp(STR, "SCALING AND SQUARING") == 0) strcpy(STR, "SS");
71  else if (strcmp(STR, "FASTSCALINGANDSQUARING") == 0) strcpy(STR, "FastSS");
72  else if (strcmp(STR, "FAST SCALING AND SQUARING") == 0) strcpy(STR, "FastSS");
73  else if (strcmp(STR, "EULER") == 0) strcpy(STR, "RKE1");
74  else if (strcmp(STR, "FORWARDEULER") == 0) strcpy(STR, "RKE1");
75  else if (strcmp(STR, "FORWARD EULER") == 0) strcpy(STR, "RKE1");
76  else if (strcmp(STR, "FWDEULER") == 0) strcpy(STR, "RKE1");
77  else if (strcmp(STR, "MODIFIEDEULER") == 0) strcpy(STR, "RKE2");
78  else if (strcmp(STR, "MODIFIED EULER") == 0) strcpy(STR, "RKE2");
79  else if (strcmp(STR, "MODEULER") == 0) strcpy(STR, "RKE2");
80  else if (strcmp(STR, "HEUN") == 0) strcpy(STR, "RKH2");
81  else if (strcmp(STR, "IMPROVEDEULER") == 0) strcpy(STR, "RKH2");
82  else if (strcmp(STR, "IMPROVED EULER") == 0) strcpy(STR, "RKH2");
83  else if (strcmp(STR, "IMPEULER") == 0) strcpy(STR, "RKH2");
84  else if (strcmp(STR, "RK1") == 0) strcpy(STR, "RKE1");
85  else if (strcmp(STR, "RK2") == 0) strcpy(STR, "RKH2");
86  else if (strcmp(STR, "RK12") == 0) strcpy(STR, "RKEH12");
87  else if (strcmp(STR, "RK23") == 0) strcpy(STR, "RKBS23");
88  else if (strcmp(STR, "RK45") == 0) strcpy(STR, "RKCK45");
89  // Convert string to enumeration value
90  m = FFDIM_NUM;
91  const char *s;
92  do {
93  m = static_cast<FFDIntegrationMethod>(static_cast<int>(m) - 1);
94  if (m == FFDIM_Unknown) break;
95  s = ToString(m);
96  } while (strcmp(s, str) != 0 && strcmp(s, STR) != 0);
97  // Clean up
98  delete[] STR;
99  // Return whether conversion was successful
100  return (m != FFDIM_Unknown);
101 }
