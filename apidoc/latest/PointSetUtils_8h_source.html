164 /// Compute orthogonal vectors spanning the tangent plane of a cell
165 ///
166 /// @param[in] n Cell normal vector.
167 /// @param[out] e1 First normal vector in tangent plane.
168 /// @param[out] e2 Second normal vector in tangent plane.
169 ///
170 /// @returns Whether tangent vectors are valid.
171 inline bool ComputeTangents(const double n[3], double e1[3], double e2[3])
172 {
173  double v[3] = {n[1], n[2], n[0]};
174  vtkMath::Cross(n, v, e1);
175  if (vtkMath::Dot(e1, e1) < 1e-6) {
176  v[1] *= -1.0;
177  vtkMath::Cross(n, v, e1);
178  if (vtkMath::Dot(e1, e1) < 1e-6) return false;
179  }
180  vtkMath::Cross(n, e1, e2);
181  vtkMath::Normalize(e1);
182  vtkMath::Normalize(e2);
183  return true;
184 }

186 // -----------------------------------------------------------------------------
187 /// Compute 4 (8) equally spaced tangent vectors (angular sampling of 45 degrees)
188 inline bool ComputeTangents(const double n[3], double e1[3], double e2[3], double e3[3], double e4[3])
189 {
190  if (!ComputeTangents(n, e1, e3)) return false;
191  const double sqrt2 = sqrt(2.0);
192  e2[0] = (e1[0] + e3[0]) / sqrt2;
193  e2[1] = (e1[1] + e3[1]) / sqrt2;
194  e2[2] = (e1[2] + e3[2]) / sqrt2;
195  e4[0] = (e1[0] - e3[0]) / sqrt2;
196  e4[1] = (e1[1] - e3[1]) / sqrt2;
197  e4[2] = (e1[2] - e3[2]) / sqrt2;
198  return true;
199 }
