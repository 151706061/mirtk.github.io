45  /// Enumeration of smoothing kernel functions
46  enum WeightFunction

66  /// Input point mask, only points with mask value != 0 are modified
67  mirtkPublicAttributeMacro(vtkSmartPointer<vtkDataArray>, Mask);
68 
69  /// Number of smoothing iterations
70  mirtkPublicAttributeMacro(int, NumberOfIterations);
71 
72  /// Relaxation / scale factor for odd iterations
73  mirtkPublicAttributeMacro(double, Lambda);
74 
75  /// Relaxation / scale factor for even iterations
76  ///
77  /// By default, when _Mu is set to NaN, a _Mu value equal to _Lambda is used.
78  /// A negative _Mu value, greater in magnitude than _Lambda, can be used
79  /// to perform a low-pass filtering of the surface mesh as described in:
80  /// Taubin, G., Curve and Surface Smoothing without Shrinkage, ICCV 1995.
81  mirtkPublicAttributeMacro(double, Mu);
82 
83  /// Smoothing kernel parameter
84  ///
85  /// In case of a Gaussian smoothing kernel, if the sigma value is negative,
86  /// the standard deviation of the Gaussian kernel is set to the average
87  /// edge length times the absolute value of \c _Sigma. If _Sigma is zero,
88  /// the average edge length is used as standard deviation. The _Sigma attribute
89  /// is set to the actual used standard deviation after the filter execution.
90  ///
91  /// In case of the inverse distance weighting, the _Sigma value is added to
92  /// the edge length before computing the inverse value.
93  mirtkPublicAttributeMacro(double, Sigma);
94 
95  /// Smoothing kernel parameter in direction of maximum curvature
96  ///
97  /// \note The direction of maximum principle curvature is orthogonal to the
98  /// direction in which the surface is most bended! It is the direction
99  /// with the most variance, i.e., along ridges, not orthogonal to these.
100  ///
101  /// This parameter is only used by the anisotropic Gaussian kernel.
102  /// By default, i.e., when _Sigma2 = 0, the standard deviation along the
103  /// direction of maximum change is half the standard deviation along the
104  /// direction of minimum change. Hence, the surface points or data values are
105  /// smoothed less in the direction of maximum change (i.e., maximum curvature).
106  /// If the sigma value is negative, the standard deviation is set to the
107  /// average edge length times the absolute value of \c _Sigma2.
108  ///
109  /// When an array of local geometry tensors is used instead of the direction
110  /// of minimum and/or maximum change, the default is to use an isotropic
111  /// Gaussian kernel in the local coordinate system defined by the tensor.
112  /// In this case the axes of the local coordinate system are expected to be
113  /// scaled anisotropically as in case of the curvature tensor, for example.
114  mirtkPublicAttributeMacro(double, MaximumDirectionSigma);
115 
116  /// Smoothing kernel function
117  mirtkPublicAttributeMacro(WeightFunction, Weighting);
118 
119  /// Name of input point data array with local geometry tensor used for anisotropic smoothing
120  ///
121  /// For example, the local curvature tensors can be computed using
122  /// PolyDataCurvature and used for anisotropic Gaussian smoothing.
123  /// The input point data array is then named PolyDataCurvature::TENSOR.
124  mirtkPublicAttributeMacro(string, GeometryTensorName);
125 
126  /// Name of input point data array with direction along which to smooth less
127  /// \note This array is only used if no _GeometryTensorName is specified.
128  mirtkPublicAttributeMacro(string, MinimumDirectionName);
129 
130  /// Name of input point data array with direction along which to smooth more
131  /// \note This array is only used if no _GeometryTensorName is specified.
132  mirtkPublicAttributeMacro(string, MaximumDirectionName);
133 
134  /// Whether to average values of adjacent nodes only or to
135  /// also include the node's values themselves in the average
136  ///
137  /// \note In case of an InverseDistance node weighting, the values of the
138  /// node itself are only included in the average if _Sigma > .0.
139  mirtkPublicAttributeMacro(bool, AdjacentValuesOnly);
140 
141  /// Whether to smooth the node positions, i.e., input geometry
142  mirtkPublicAttributeMacro(bool, SmoothPoints);
143 
144  /// Names of input point data arrays to be smoothed
145  mirtkPublicAttributeMacro(ArrayNames, SmoothArrays);
146 
147  /// Input point data arrays to be smoothed
148  mirtkAttributeMacro(DataArrays, InputArrays);
149 
150  /// Output point data arrays
151  mirtkAttributeMacro(DataArrays, OutputArrays);
152 
153  /// Verbosity of output messages
154  mirtkPublicAttributeMacro(int, Verbose);
