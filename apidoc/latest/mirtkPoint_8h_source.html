587 // -----------------------------------------------------------------------------
588 inline Point Point::operator* (const Matrix& m) const
589 {
590  Point tmp;
591  if ((m.Rows() != 4) && (m.Cols() != 4)) {
592  cerr << "Point::operator*(const Matrix& m): Size mismatch" << endl;
593  exit(1);
594  }
595  tmp._x = _x * m(0, 0) + _y * m(0, 1) + _z * m(0, 2) + m(0, 3);
596  tmp._y = _x * m(1, 0) + _y * m(1, 1) + _z * m(1, 2) + m(1, 3);
597  tmp._z = _x * m(2, 0) + _y * m(2, 1) + _z * m(2, 2) + m(2, 3);
598  return tmp;
599 }

601 // -----------------------------------------------------------------------------
602 inline Point& Point::operator*=(const Matrix& m)
603 {
604  Point tmp;
605  if ((m.Rows() != 4) && (m.Cols() != 4)) {
606  cerr << "Point::operator*(const Matrix& m): Size mismatch" << endl;
607  exit(1);
608  }
609  tmp._x = _x * m(0, 0) + _y * m(0, 1) + _z * m(0, 2) + m(0, 3);
610  tmp._y = _x * m(1, 0) + _y * m(1, 1) + _z * m(1, 2) + m(1, 3);
611  tmp._z = _x * m(2, 0) + _y * m(2, 1) + _z * m(2, 2) + m(2, 3);
612  *this = tmp;
613  return *this;
614 }

623 inline double Point::SquaredDistance(const Point &p) const
