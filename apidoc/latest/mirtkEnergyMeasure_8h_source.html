105 /// Convert energy measure enumeration value to string
106 inline string ToString(const EnergyMeasure &em)
107 {
108  switch (em) {
109  // ---------------------------------------------------------------------------
110  // Image (dis-)similarity measures
111  case EM_JE: return "JE";
112  case EM_CC: return "CC";
113  case EM_MI: return "MI";
114  case EM_NMI: return "NMI";
115  case EM_SSD: return "SSD";
116  case EM_CR_XY: return "CR_XY";
117  case EM_CR_YX: return "CR_YX";
118  case EM_LC: return "LC";
119  case EM_K: return "K";
120  case EM_ML: return "ML";
121  case EM_NGF_COS: return "NGF_COS";
122  case EM_LNCC: return "LNCC";
123 
124  // ---------------------------------------------------------------------------
125  // Point set distance measures
126  case EM_FRE: return "FRE";
127  case EM_CorrespondenceDistance: return "PCD";
128  case EM_CurrentsDistance: return "CurrentsDistance";
129  case EM_VarifoldDistance: return "VarifoldDistance";
130 
131  // ---------------------------------------------------------------------------
132  // External point set forces
133  case EM_BalloonForce: return "BalloonForce";
134  case EM_ImageEdgeForce: return "ImageEdgeForce";
135  case EM_ImplicitSurfaceDistance: return "ImplicitSurfaceDistance";
136 
137  // ---------------------------------------------------------------------------
138  // Internal point set forces
139  case EM_MetricDistortion: return "MetricDistortion";
140  case EM_Stretching: return "Stretching";
141  case EM_Curvature: return "Curvature";
142  case EM_NonSelfIntersection: return "NSI";
143  case EM_RepulsiveForce: return "Repulsion";
144  case EM_InflationForce: return "Inflation";
145 
146  // -------------------------------------------------------------------------
147  // Transformation constraints
148  case EM_BendingEnergy: return "BE";
149  case EM_VolumePreservation: return "VP";
150  case EM_TopologyPreservation: return "TP";
151  case EM_Sparsity: return "Sparsity";
152  case EM_L0Norm: return "L0";
153  case EM_L1Norm: return "L1";
154  case EM_L2Norm: return "L2";
155  case EM_SqLogDetJac: return "SqLogDetJac";
156  case EM_MinDetJac: return "MinDetJac";
157 
158  // ---------------------------------------------------------------------------
159  // Unknown/invalid enumeration value
160  default: return "Unknown";
161  }
162 }

165 /// Convert energy measure string to enumeration value
166 inline bool FromString(const char *str, EnergyMeasure &em)
167 {
168  em = EM_Unknown;
169 
170  // ---------------------------------------------------------------------------
171  // Alternative names for image (dis-)similarity measures
172  if (em == EM_Unknown) {
173  if (strcmp(str, "NCC") == 0) em = EM_LNCC;
174  else if (strcmp(str, "LCC") == 0) em = EM_LNCC;
175  }
176 
177  // ---------------------------------------------------------------------------
178  // Alternative names for point set distance measures
179  if (em == EM_Unknown) {
180  if (strcmp(str, "Fiducial Registration Error") == 0 ||
181  strcmp(str, "Fiducial registration error") == 0 ||
182  strcmp(str, "Fiducial Error") == 0 ||
183  strcmp(str, "Fiducial error") == 0 ||
184  strcmp(str, "Landmark Registration Error") == 0 ||
185  strcmp(str, "Landmark registration error") == 0 ||
186  strcmp(str, "Landmark Error") == 0 ||
187  strcmp(str, "Landmark error") == 0) {
188  em = EM_FRE;
189  } else if (strcmp(str, "Point Correspondence Distance") == 0 ||
190  strcmp(str, "Point correspondence distance") == 0 ||
191  strcmp(str, "Correspondence Distance") == 0 ||
192  strcmp(str, "Correspondence distance") == 0) {
193  em = EM_CorrespondenceDistance;
194  } else if (strcmp(str, "Currents distance") == 0 ||
195  strcmp(str, "Currents Distance") == 0) {
196  em = EM_CurrentsDistance;
197  } else if (strcmp(str, "Varifold distance") == 0 ||
198  strcmp(str, "Varifold Distance") == 0) {
199  em = EM_VarifoldDistance;
200  }
201  }
202 
203  // ---------------------------------------------------------------------------
204  // Alternative names for external point set forces
205  if (em == EM_Unknown) {
206  if (strcmp(str, "EdgeForce") == 0) em = EM_ImageEdgeForce;
207  }
208 
209  // ---------------------------------------------------------------------------
210  // Alternative names for internal point set forces
211  if (em == EM_Unknown) {
212  if (strcmp(str, "EdgeLength") == 0) em = EM_Stretching;
213  else if (strcmp(str, "MetricDistortion") == 0) em = EM_MetricDistortion;
214  else if (strcmp(str, "Bending") == 0) em = EM_Curvature;
215  else if (strcmp(str, "SurfaceBending") == 0) em = EM_Curvature;
216  else if (strcmp(str, "SurfaceCurvature") == 0) em = EM_Curvature;
217  else if (strcmp(str, "RepulsiveForce") == 0) em = EM_RepulsiveForce;
218  else if (strcmp(str, "NonSelfIntersection") == 0) em = EM_NonSelfIntersection;
219  else if (strcmp(str, "InflationForce") == 0) em = EM_InflationForce;
220  else if (strcmp(str, "SurfaceInflation") == 0) em = EM_InflationForce;
221  }
222 
223  // ---------------------------------------------------------------------------
224  // Alternative names for transformation regularization terms
225  if (em == EM_Unknown) {
226  if (strcmp(str, "JAC") == 0) em = EM_SqLogDetJac;
227  else if (strcmp(str, "MinJac") == 0) em = EM_MinDetJac;
228  }
229 
230  // ---------------------------------------------------------------------------
231  // Convert default names of energy measures (cf. ToString(EnergyMeasure))
232  if (em == EM_Unknown) {
233  em = static_cast<EnergyMeasure>(EM_Last - 1);
234  while (em != EM_Unknown) {
235  if (ToString(em) == str) break;
236  em = static_cast<EnergyMeasure>(em - 1);
237  }
238  }
239 
240  return (em != EM_Unknown);
241 }
