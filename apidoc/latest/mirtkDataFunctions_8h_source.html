32 /// Set value of unmasked data points
33 class SetInsideValue : public Op
34 {
35 private:
36 
37  double *_Data;
38  bool *_Mask;
39  double _Value;
40 
41 public:
42 
43  SetInsideValue(double value = .0) : _Value(value) {}
44 
45  // Called by TBB's parallel_for, for internal use only!
46  void operator()(const blocked_range<int> &re) const
47  {
48  double *data = _Data + re.begin();
49  if (_Mask) {
50  bool *mask = _Mask + re.begin();
51  for (int i = re.begin(); i != re.end(); ++i, ++data, ++mask) {
52  if (*mask == true) *data = _Value;
53  }
54  } else {
55  for (int i = re.begin(); i != re.end(); ++i, ++data) {
56  *data = _Value;
57  }
58  }
59  }
60 
61  /// Process given data (not thread-safe!)
62  virtual void Process(int n, double *data, bool *mask = NULL)
63  {
64  _Data = data;
65  _Mask = mask;
66  parallel_for(blocked_range<int>(0, n), *this);
67  }
68 };

106 /// Base class of element-wise data transformations
107 class ElementWiseUnaryOp : public Op
108 {
109 private:
110 
111  double *_Data;
112  bool *_Mask;
113 
114 public:
115 
116  // Called by TBB's parallel_for, for internal use only!
117  void operator()(const blocked_range<int> &re) const
118  {
119  double *data = _Data + re.begin();
120  if (_Mask) {
121  bool *mask = _Mask + re.begin();
122  for (int i = re.begin(); i != re.end(); ++i, ++data, ++mask) {
123  if (*mask) *data = this->Op(*data, *mask);
124  }
125  } else {
126  bool mask = true;
127  for (int i = re.begin(); i != re.end(); ++i, ++data) {
128  *data = this->Op(*data, mask);
129  }
130  }
131  }
132 
133  /// Transform data value and/or mask it by setting mask = false
134  virtual double Op(double value, bool &) const = 0;
135 
136  /// Process given data (not thread-safe!)
137  virtual void Process(int n, double *data, bool *mask = NULL)
138  {
139  _Data = data;
140  _Mask = mask;
141  // MUST be called in the base class which defines Op!
142  //parallel_for(blocked_range<int>(0, n), *this);
143  }

147 /// Base class of element-wise data transformations
148 class ElementWiseBinaryOp : public Op
149 {
150  /// Constant value to add
151  mirtkPublicAttributeMacro(double, Constant);
152 
153  /// File path of second dataset
154  mirtkPublicAttributeMacro(string, FileName);
155 
156 private:
157 
158  double *_Data;
159  bool *_Mask;
160  double *_Other;
161 
162 protected:
163 
164  /// Constructor
165  ElementWiseBinaryOp(double value) : _Constant(value) {}
166 
167  /// Constructor
168  ElementWiseBinaryOp(const char *fname) : _Constant(.0), _FileName(fname) {}
169 
170 public:
171 
172  // Called by TBB's parallel_for, for internal use only!
173  void operator()(const blocked_range<int> &re) const
174  {
175  double *data = _Data + re.begin();
176  if (_Other) {
177  double *other = _Other + re.begin();
178  if (_Mask) {
179  bool *mask = _Mask + re.begin();
180  for (int i = re.begin(); i != re.end(); ++i) {
181  if (*mask) *data = this->Op(*data, *other, *mask);
182  ++data, ++other, ++mask;
183  }
184  } else {
185  bool mask = true;
186  for (int i = re.begin(); i != re.end(); ++i) {
187  *data = this->Op(*data, *other, mask);
188  ++data, ++other;
189  }
190  }
191  } else {
192  if (_Mask) {
193  bool *mask = _Mask + re.begin();
194  for (int i = re.begin(); i != re.end(); ++i) {
195  if (*mask) *data = this->Op(*data, _Constant, *mask);
196  ++data, ++mask;
197  }
198  } else {
199  bool mask = true;
200  for (int i = re.begin(); i != re.end(); ++i) {
201  *data = this->Op(*data, _Constant, mask);
202  ++data;
203  }
204  }
205  }
206  }
207 
208  /// Transform data value and/or mask it by setting mask = false
209  virtual double Op(double value, double, bool &) const = 0;
210 
211  /// Process given data (not thread-safe!)
212  virtual void Process(int n, double *data, bool *mask = NULL)
213  {
214  _Data = data;
215  _Mask = mask;
216  _Other = NULL;
217  if (!_FileName.empty()) {
218  if (Read(_FileName.c_str(), _Other) != n) {
219  cerr << "Input file " << _FileName << " has different number of data points!" << endl;
220  exit(1);
221  }
222  }
223  // MUST be called in the base class which defines Op!
224  //parallel_for(blocked_range<int>(0, n), *this);
225  }
226 };

818 /// Rescale values to new range
819 class Rescale : public Op
820 {
821 private:
822 
823  double *_Data;
824  bool *_Mask;
825  double _Min;
826  double _Max;
827  double _Slope;
828  double _Intercept;
829 
830 public:
831 
832  Rescale(double min = .0, double max = 1.0) : _Min(min), _Max(max) {}
833 
834  // Called by TBB's parallel_for, for internal use only!
835  void operator()(const blocked_range<int> &re) const
836  {
837  double *data = _Data + re.begin();
838  for (int i = re.begin(); i != re.end(); ++i, ++data) {
839  *data = (*data * _Slope) + _Intercept;
840  }
841  }
842 
843  /// Process given data (not thread-safe!)
844  virtual void Process(int n, double *data, bool *mask = NULL)
845  {
846  statistic::Extrema extrema;
847  extrema.Evaluate(n, data, mask);
848  _Slope = (_Max - _Min) / (extrema.Max() - extrema.Min());
849  _Intercept = _Min - (extrema.Min() * _Slope);
850  _Data = data;
851  _Mask = mask;
852  parallel_for(blocked_range<int>(0, n), *this);
853  }
854 };
