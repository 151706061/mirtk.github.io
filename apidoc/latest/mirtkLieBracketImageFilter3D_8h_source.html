124 // ---------------------------------------------------------------------------
125 // Note: Using NN extrapolation at boundary
126 template <class VoxelType>
127 void
128 LieBracketImageFilter3D<VoxelType>
129 ::Jacobian(Matrix &jac, const GenericImage<VoxelType> &v, int i, int j, int k)
130 {
131  int a, b;
132  // Finite difference in x dimension
133  if (i <= 0) {
134  a = i;
135  b = i + 1;
136  } else if (i >= v.GetX()-1) {
137  a = i - 1;
138  b = i;
139  } else {
140  a = i - 1;
141  b = i + 1;
142  }
143  jac(0, 0) = 0.5 * (v(b, j, k, 0) - v(a, j, k, 0));
144  jac(1, 0) = 0.5 * (v(b, j, k, 1) - v(a, j, k, 1));
145  jac(2, 0) = 0.5 * (v(b, j, k, 2) - v(a, j, k, 2));
146  // Finite difference in y dimension
147  if (j <= 0) {
148  a = j;
149  b = j + 1;
150  } else if (j >= v.GetY()-1) {
151  a = j - 1;
152  b = j;
153  } else {
154  a = j - 1;
155  b = j + 1;
156  }
157  jac(0, 1) = 0.5 * (v(i, b, k, 0) - v(i, a, k, 0));
158  jac(1, 1) = 0.5 * (v(i, b, k, 1) - v(i, a, k, 1));
159  jac(2, 1) = 0.5 * (v(i, b, k, 2) - v(i, a, k, 2));
160  // Finite difference in z dimension
161  if (k <= 0) {
162  a = k;
163  b = k + 1;
164  } else if(k >= v.GetZ()-1) {
165  a = k - 1;
166  b = k;
167  } else {
168  a = k - 1;
169  b = k + 1;
170  }
171  jac(0, 2) = 0.5 * (v(i, j, b, 0) - v(i, j, a, 0));
172  jac(1, 2) = 0.5 * (v(i, j, b, 1) - v(i, j, a, 1));
173  jac(2, 2) = 0.5 * (v(i, j, b, 2) - v(i, j, a, 2));
174  // Project derivative from image to world space
175  jac *= _MatW2I;
176 }

179 template <class VoxelType>
180 double LieBracketImageFilter3D<VoxelType>::Run(int i, int j, int k, int t)
181 {
182  Matrix lJ(3, 3), rJ(3, 3);
183 
184  const GenericImage<VoxelType> &lv = *this->GetInput(0);
185  const GenericImage<VoxelType> &rv = *this->GetInput(1);
186 
187  const VoxelType &lx = lv(i, j, k, 0);
188  const VoxelType &ly = lv(i, j, k, 1);
189  const VoxelType &lz = lv(i, j, k, 2);
190  const VoxelType &rx = rv(i, j, k, 0);
191  const VoxelType &ry = rv(i, j, k, 1);
192  const VoxelType &rz = rv(i, j, k, 2);
193 
194  Jacobian(lJ, lv, i, j, k);
195  Jacobian(rJ, rv, i, j, k);
196 
197  return (lJ(t, 0) * rx - lx * rJ(t, 0)) +
198  (lJ(t, 1) * ry - ly * rJ(t, 1)) +
199  (lJ(t, 2) * rz - lz * rJ(t, 2));
200 }

203 template <class VoxelType>
204 void LieBracketImageFilter3D<VoxelType>::Run(double vec[3], int i, int j, int k)
205 {
206  Matrix lJ(3, 3), rJ(3, 3);
207 
208  const GenericImage<VoxelType> &lv = *this->GetInput(0);
209  const GenericImage<VoxelType> &rv = *this->GetInput(1);
210 
211  const VoxelType &lx = lv(i, j, k, 0);
212  const VoxelType &ly = lv(i, j, k, 1);
213  const VoxelType &lz = lv(i, j, k, 2);
214  const VoxelType &rx = rv(i, j, k, 0);
215  const VoxelType &ry = rv(i, j, k, 1);
216  const VoxelType &rz = rv(i, j, k, 2);
217 
218  Jacobian(lJ, lv, i, j, k);
219  Jacobian(rJ, rv, i, j, k);
220 
221  vec[0] = (lJ(0, 0) * rx - lx * rJ(0, 0)) +
222  (lJ(0, 1) * ry - ly * rJ(0, 1)) +
223  (lJ(0, 2) * rz - lz * rJ(0, 2));
224 
225  vec[1] = (lJ(1, 0) * rx - lx * rJ(1, 0)) +
226  (lJ(1, 1) * ry - ly * rJ(1, 1)) +
227  (lJ(1, 2) * rz - lz * rJ(1, 2));
228 
229  vec[2] = (lJ(2, 0) * rx - lx * rJ(2, 0)) +
230  (lJ(2, 1) * ry - ly * rJ(2, 1)) +
231  (lJ(2, 2) * rz - lz * rJ(2, 2));
232 }

234 // ---------------------------------------------------------------------------
235 /// Parallelizable body of LieBracketImageFilter3D::Run.
236 template <class VoxelType>
237 class LieBracketImageFilter3DRun
238 {
239 private:
240  LieBracketImageFilter3D<VoxelType> *_LieBracketFilter; ///< Lie bracket filter
241  GenericImage<VoxelType> *_Output; ///< Output vector field
242 
243 public:
244 
245  /// Default constructor
246  LieBracketImageFilter3DRun(LieBracketImageFilter3D<VoxelType> *filter,
247  GenericImage<VoxelType> *output)
248  :
249  _LieBracketFilter(filter),
250  _Output(output)
251  {
252  }
253 
254  /// Copy constructor
255  LieBracketImageFilter3DRun(const LieBracketImageFilter3DRun<VoxelType> &other)
256  :
257  _LieBracketFilter(other._LieBracketFilter),
258  _Output (other._Output)
259  {
260  }
261 
262  /// Run Lie bracket filter for each voxel in specified range
263  void operator ()(const blocked_range3d<int> &r) const
264  {
265  double vec[3];
266  for (int k = r.pages().begin(); k != r.pages().end(); k++) {
267  for (int j = r.rows().begin(); j != r.rows().end(); j++) {
268  for (int i = r.cols().begin(); i != r.cols().end(); i++) {
269  _LieBracketFilter->Run(vec, i, j, k);
270  _Output->Put(i, j, k, 0, vec[0]);
271  _Output->Put(i, j, k, 1, vec[1]);
272  _Output->Put(i, j, k, 2, vec[2]);
273  }
274  }
275  }
276  }
277 };

280 template <class VoxelType>
281 void LieBracketImageFilter3D<VoxelType>::Run()
282 {
283  blocked_range3d<int> voxels(0, this->GetInput(0)->Z(), 1,
284  0, this->GetInput(0)->Y(), 1,
285  0, this->GetInput(0)->X(), 1);
286  LieBracketImageFilter3DRun<VoxelType> body(this, this->Output());
287  MIRTK_START_TIMING();
288  this->Initialize();
289  parallel_for(voxels, body);
290  this->Finalize();
291  MIRTK_DEBUG_TIMING(2, "LieBracketImageFilter");
292 }
