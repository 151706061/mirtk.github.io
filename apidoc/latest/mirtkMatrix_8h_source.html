689  }
690  return true;
691 }

840 inline double Matrix::Trace() const
841 {
842  if (_rows == _cols) {
843  double trace = 0;
844  for (int j = 0; j < _cols; ++j)
845  for (int i = 0; i < _rows; ++i) {
846  trace += _matrix[j][i];
847  }
848  return trace;
849  } else {
850  cerr << "Matrix::Trace() matrix number of col != row" << endl;
851  return 0;
852  }
853 }

855 // -----------------------------------------------------------------------------
856 inline double Matrix::Norm() const
857 {
858  int i, j;
859  double norm = 0;
860 
861  // The norm of a matrix M is defined as trace(M * M~)
862  for (j = 0; j < _cols; j++) {

872 {
873  int i, j;
874  double normInf = -1.0 * DBL_MAX;
875  double sum;
876 
877  for (i = 0; i < _rows; ++i) {
878  sum = 0;
879  for (j = 0; j < _cols; ++j) {
880  sum += abs(_matrix[j][i]);
881  }

883  normInf = sum;
884  }
