31 /**
32  * Volume preservation constraint for deformable image registration
33  *
34  * The volume preservation constraint is based on a penalty imposed by the
35  * squared logarithm of the Jacobian determinant. Unlike the base class
36  * implementation which computes the Jacobian w.r.t the spline function,
37  * this constraint computes the Jacobian always w.r.t the transformed
38  * coordinates. These two derivative computations are identical for a
39  * FFD parameterized by control point displacements, but differ in case of
40  * a stationary velocity field parameterization.
41  *
42  * Torsten RohlÔ¨Ång and Calvin R. Maurer, Jr., Intensity-Based Non-rigid
43  * Registration Using Adaptive Multilevel Free-Form Deformation with an
44  * Incompressibility Constraint, MICCAI 2001.
45  */
46 class VolumePreservationConstraint : public LogJacobianConstraint
47 {
48  mirtkEnergyTermMacro(VolumePreservationConstraint, EM_VolumePreservation);
49 
50 public:
51 
52  /// Constructor
53  VolumePreservationConstraint(const char * name = "")
54  :
55  LogJacobianConstraint(name)
56  {}
57 
58  /// Destructor
59  virtual ~VolumePreservationConstraint() {}
60 
61  /// Compute determinant and adjugate of Jacobian of transformation
62  virtual double Jacobian(const FreeFormTransformation *ffd,
63  double x, double y, double z, double t,
64  Matrix &adj) const
65  {
66  double det;
67  ffd->LocalJacobian(adj, x, y, z, t, t);
68  adj.Adjugate(det);
69  if (det < 1e-7) det = 1e-7;
70  return det;
71  }
72 
73 };
