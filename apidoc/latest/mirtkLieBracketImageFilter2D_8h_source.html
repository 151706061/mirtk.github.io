121 // ---------------------------------------------------------------------------
122 // Note: Using NN extrapolation at boundary
123 template <class VoxelType>
124 void
125 LieBracketImageFilter2D<VoxelType>
126 ::Jacobian(Matrix &jac, const GenericImage<VoxelType> &v, int i, int j)
127 {
128  int a, b;
129  // Finite difference in x dimension
130  if (i <= 0) {
131  a = i;
132  b = i + 1;
133  } else if (i >= v.X() - 1) {
134  a = i - 1;
135  b = i;
136  } else {
137  a = i - 1;
138  b = i + 1;
139  }
140  jac(0, 0) = 0.5 * (v(b, j, 0, 0) - v(a, j, 0, 0));
141  jac(1, 0) = 0.5 * (v(b, j, 0, 1) - v(a, j, 0, 1));
142  // Finite difference in y dimension
143  if (j <= 0) {
144  a = j;
145  b = j + 1;
146  } else if (j >= v.Y() - 1) {
147  a = j - 1;
148  b = j;
149  } else {
150  a = j - 1;
151  b = j + 1;
152  }
153  jac(0, 1) = 0.5 * (v(i, b, 0, 0) - v(i, a, 0, 0));
154  jac(1, 1) = 0.5 * (v(i, b, 0, 1) - v(i, a, 0, 1));
155  // Project derivative from image to world space
156  jac *= _MatW2I;
157 }

159 // ---------------------------------------------------------------------------
160 template <class VoxelType>
161 double LieBracketImageFilter2D<VoxelType>::Run(int i, int j, int, int t)
162 {
163  Matrix lJ(2, 2), rJ(2, 2);
164 
165  const GenericImage<VoxelType> &lv = *this->GetInput(0);
166  const GenericImage<VoxelType> &rv = *this->GetInput(1);
167 
168  const VoxelType &lx = lv(i, j, 0, 0);
169  const VoxelType &ly = lv(i, j, 0, 1);
170  const VoxelType &rx = rv(i, j, 0, 0);
171  const VoxelType &ry = rv(i, j, 0, 1);
172 
173  Jacobian(lJ, lv, i, j);
174  Jacobian(rJ, rv, i, j);
175 
176  return (lJ(t, 0) * rx - lx * rJ(t, 0)) + (lJ(t, 1) * ry - ly * rJ(t, 1));
177 }

179 // ---------------------------------------------------------------------------
180 template <class VoxelType>
181 void LieBracketImageFilter2D<VoxelType>::Run(double vec[2], int i, int j)
182 {
183  Matrix lJ(2, 2), rJ(2, 2);
184 
185  const GenericImage<VoxelType> &lv = *this->GetInput(0);
186  const GenericImage<VoxelType> &rv = *this->GetInput(1);
187 
188  const VoxelType &lx = lv(i, j, 0, 0);
189  const VoxelType &ly = lv(i, j, 0, 1);
190  const VoxelType &rx = rv(i, j, 0, 0);
191  const VoxelType &ry = rv(i, j, 0, 1);
192 
193  Jacobian(lJ, lv, i, j);
194  Jacobian(rJ, rv, i, j);
195 
196  vec[0] = (lJ(0, 0) * rx - lx * rJ(0, 0)) + (lJ(0, 1) * ry - ly * rJ(0, 1));
197  vec[1] = (lJ(1, 0) * rx - lx * rJ(1, 0)) + (lJ(1, 1) * ry - ly * rJ(1, 1));
198 }

201 /// Parallelizable body of LieBracketImageFilter2D::Run.
202 template <class VoxelType>
203 class LieBracketImageFilter2DRun
204 {
205 private:
206  LieBracketImageFilter2D<VoxelType> *_LieBracketFilter; ///< Lie bracket filter
207  GenericImage<VoxelType> *_Output; ///< Output vector field
208 
209 public:
210 
211  /// Default constructor
212  LieBracketImageFilter2DRun(LieBracketImageFilter2D<VoxelType> *filter,
213  GenericImage<VoxelType> *output)
214  :
215  _LieBracketFilter(filter),
216  _Output(output)
217  {
218  }
219 
220  /// Copy constructor
221  LieBracketImageFilter2DRun(const LieBracketImageFilter2DRun<VoxelType> &other)
222  :
223  _LieBracketFilter(other._LieBracketFilter),
224  _Output (other._Output)
225  {
226  }
227 
228  /// Run Lie bracket filter for each voxel in specified range
229  void operator ()(const blocked_range2d<int> &r) const
230  {
231  double vec[2];
232  for (int j = r.rows().begin(); j != r.rows().end(); j++) {
233  for (int i = r.cols().begin(); i != r.cols().end(); i++) {
234  _LieBracketFilter->Run(vec, i, j);
235  _Output->Put(i, j, 0, 0, vec[0]);
236  _Output->Put(i, j, 0, 1, vec[1]);
237  }
238  }
239  }
240 };

242 // ---------------------------------------------------------------------------
243 template <class VoxelType>
244 void LieBracketImageFilter2D<VoxelType>::Run()
245 {
246  blocked_range2d<int> voxels(0, this->GetInput(0)->Y(), 1,
247  0, this->GetInput(0)->X(), 1);
248  LieBracketImageFilter2DRun<VoxelType> body(this, this->Output());
249  MIRTK_START_TIMING();
250  this->Initialize();
251  parallel_for(voxels, body);
252  this->Finalize();
253  MIRTK_DEBUG_TIMING(2, "LieBracketImageFilter");
254 }
