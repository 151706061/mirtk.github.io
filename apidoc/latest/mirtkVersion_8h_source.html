104 // -----------------------------------------------------------------------------
105 /// Write version to output stream
106 inline ostream &operator <<(ostream &os, const Version &version)
107 {
108  os << version.Major() << "." << version.Minor();
109  if (version.Patch() != 0u) os << "." << version.Patch();
110  return os;
111 }
112 
113 // -----------------------------------------------------------------------------
114 /// Read version from input stream
115 inline istream &operator >>(istream &is, Version &version)
116 {
117  string token;
118  char c;
119  while (is.get(c) && (isdigit(c) || c == '.')) token.push_back(c);
120  if (is.bad()) return is;
121  if (is.eof()) is.clear(ios::eofbit); // reset failbit
122  istringstream ss(token);
123  unsigned int number[3] = {0u, 0u, 0u};
124  int i = -1;
125  while (getline(ss, token, '.')) {
126  if (++i == 3 || !FromString(token.c_str(), number[i])) {
127  i = -1;
128  break;
129  }
130  }
131  if (i == -1) is.setstate(ios::failbit);
132  else {
133  version.Major(number[0]);
134  version.Minor(number[1]);
135  version.Patch(number[2]);
136  }
137  return is;
138 }
