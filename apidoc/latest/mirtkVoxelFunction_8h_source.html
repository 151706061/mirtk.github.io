40 /**
41  * Base class for voxel functions
42  *
43  * The basic voxel functions which are part of this library and included by the
44  * VoxelFunction.h module demonstrate the implementation and use of the
45  * ForEachVoxel function templates. These templates can be well optimized by the
46  * compiler and provide the fastest way of iterating over all voxels in a given
47  * image region (or the entire image) of one or more images. When processing
48  * multiple images at the same time, these must have the same image size.
49  *
50  * Example usage:
51  * \code
52  * // Determine intensity range of grayscale image
53  * GreyImage image(attr);
54  * UnaryVoxelFunction::GetMinMax minmax;
55  * ForEachVoxel(image, minmax);
56  * double min = minmax.GetMinAsDouble();
57  * double max = minmax.GetMaxAsDouble();
58  *
59  * // Add three images within a given region and store result in another
60  * NaryVoxelFunction::Sum sum;
61  * GreyImage input1(attr);
62  * GreyImage input2(attr);
63  * GreyImage input3(attr);
64  * GreyImage output(attr);
65  * blocked_range3d<int> region(1, 2, 0, attr._y, 0, attr._x);
66  * ForEachVoxel(region, input1, input2, input3, output, sum);
67  *
68  * // Subtract one image from another
69  * // Note: Voxel function passed in this case by copy instead of reference!
70  * // This is only possible if it is not a voxel reduction.
71  * GreyImage minuend (attr);
72  * GreyImage subtrahend(attr);
73  * ParallelForEachVoxel(BinaryVoxelFunction::Sub(), subtrahend, minuend);
74  * \endcode
75  *
76  * If a voxel function needs to consider also the intensities of neighboring
77  * voxels, it can access these by manipulating the given pointer to the current
78  * voxel to be processed. A useful helper for this is the ImageRegion class.
79  *
80  * \sa ImageRegion, ForEachVoxelDomain
81  */
82 struct VoxelFunction
83 {
84  /// Used by ParallelForEachVoxel to determine if voxel function has to
85  /// be executed using parallel_reduce
86  static bool IsReduction() { return false; }
87 
88  /// Split "constructor"
89  ///
90  /// \note A method is used instead of an actual split constructor such that
91  /// subclasses which are not used for reduction do not need to
92  /// implement such constructor.
93  void split(VoxelFunction &) {}
94 
95  /// Join results
96  void join(VoxelFunction &) {}
97 };

100 /**
101  * Base class for voxel functions which implement a reduction of voxel values
102  *
103  * Voxel functions derived from this base class are run by ParallelForEachVoxel
104  * using parallel_reduce instead of parallel_for. Typical voxel reduction
105  * functions are unary functions computing image statistics such as the min,
106  * max, mean, or variance of voxel values within an image region.
107  *
108  */
109 struct VoxelReduction : public VoxelFunction
110 {
111  /// Used by ParallelForEachVoxel to determine if voxel function has to
112  /// be executed using parallel_reduce
113  static bool IsReduction() { return true; }
114 
115  /// Split "constructor"
116  void split(VoxelFunction &)
117  {
118  cerr << "VoxelReduction::split must be overriden by each subclass!" << endl;
119  cerr << "Otherwise you should use VoxelFunction with parallel_for instead." << endl;
120  exit(1);
121  }
122 
123  /// Join results
124  void join(VoxelFunction &)
125  {
126  cerr << "VoxelReduction::join must be overriden by each subclass!" << endl;
127  cerr << "Otherwise you should use VoxelFunction with parallel_for instead." << endl;
128  exit(1);
129  }
130 };

136 /**
137  * Base class for ForEachVoxel template function body with single voxel
138  * function for each voxel
139  */
140 template <class VoxelFunc>
141 struct ForEachVoxelBody
142 {
143  // ---------------------------------------------------------------------------
144  // Members
145 
146  VoxelFunc _VoxelFunc; ///< Functor executed for each voxel
147  int _k, _l; ///< Indices for fixed dimensions
148 
149  // ---------------------------------------------------------------------------
150  // Construction
151 
152  /// Constructor
153  ForEachVoxelBody(const VoxelFunc &vf)
154  :
155  _VoxelFunc(vf), _k(0), _l(0)
156  {}
157 
158  /// Copy constructor
159  ForEachVoxelBody(const ForEachVoxelBody &o)
160  :
161  _VoxelFunc(o._VoxelFunc), _k(o._k), _l(o._l)
162  {}
163 
164  // ---------------------------------------------------------------------------
165  // Parallel reduction
166 
167  /// Split constructor
168  ForEachVoxelBody(ForEachVoxelBody &o, split s)
169  :
170  _VoxelFunc(o._VoxelFunc), _k(o._k), _l(o._l)
171  {
172  _VoxelFunc.split(o._VoxelFunc);
173  }
174 
175  /// Join results
176  void join(ForEachVoxelBody &rhs)
177  {
178  _VoxelFunc.join(rhs._VoxelFunc);
179  }
180 };

186 /**
187  * Base class for ForEachVoxelIf template function body with separate voxel
188  * function for inside and outside voxels
189  */
190 template <class VoxelFunc, class OutsideFunc>
191 struct ForEachVoxelIfBody : public ForEachVoxelBody<VoxelFunc>
192 {
193  // ---------------------------------------------------------------------------
194  // Members
195 
196  OutsideFunc _OutsideFunc; ///< Functor executed for each background voxel
197 
198  // ---------------------------------------------------------------------------
199  // Construction
200 
201  /// Constructor
202  ForEachVoxelIfBody(const VoxelFunc &vf, const OutsideFunc &of)
203  :
204  ForEachVoxelBody<VoxelFunc>(vf), _OutsideFunc(of)
205  {}
206 
207  /// Copy constructor
208  ForEachVoxelIfBody(const ForEachVoxelIfBody &o)
209  :
210  ForEachVoxelBody<VoxelFunc>(o), _OutsideFunc(o._OutsideFunc)
211  {}
212 
213  // ---------------------------------------------------------------------------
214  // Parallel reduction
215 
216  /// Split constructor
217  ForEachVoxelIfBody(ForEachVoxelIfBody &o, split s)
218  :
219  ForEachVoxelBody<VoxelFunc>(o, s),
220  _OutsideFunc(o._OutsideFunc)
221  {
222  _OutsideFunc.split(o._OutsideFunc);
223  }
224 
225  /// Join results
226  void join(ForEachVoxelIfBody &rhs)
227  {
228  ForEachVoxelBody<VoxelFunc>::join(rhs);
229  _OutsideFunc.join(rhs._OutsideFunc);
230  }
231 };
