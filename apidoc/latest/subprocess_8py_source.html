21 This module can be used to execute the MIRTK command-line tools from an image

23 module which is used internally by this mirtk.subprocess module.

38 config = 'Release'

48 # ----------------------------------------------------------------------------
49 def path(argv, quiet=False):
50  """Get full path of MIRTK command executable."""
51  if isinstance(argv, str): command = argv
52  else: command = argv[0]
53  if command == 'convert-dof2csv': command = 'convert-dof'
54  elif command == 'convert-dof2velo': command = 'convert-dof'
55  fpath = None
56  for ext in libexec_ext:
57  p = os.path.join(libexec_dir, config, command + ext)
58  if os.path.isfile(p):
59  fpath = p
60  break
61  if not fpath:
62  for ext in libexec_ext:
63  p = os.path.join(libexec_dir, command + ext)
64  if os.path.isfile(p):
65  fpath = p
66  break
67  if not quiet:
68  if not fpath:
69  sys.stderr.write('Error: Missing execuable for command: ' + command)
70  elif not os.access(fpath, os.X_OK):
71  sys.stderr.write('Error: Insufficient permissions to execute command: ' + fpath)
72  return fpath

75 def _call(argv, verbose=0, execfunc=subprocess.call):
76  """Execute MIRTK command."""
77  if argv[0] == 'convert-dof2csv':
78  new_argv = ['convert-dof']
79  out_fmt = 'star_ccm_table'
80  for arg in argv[1:]:
81  if arg == '-pos': out_fmt = 'star_ccm_table_xyz'
82  else: new_argv.append(arg)
83  new_argv.extend(['-output-format', out_fmt])
84  argv = new_argv
85  argv[0] = path(argv)
86  if not argv[0]: return 1
87  if verbose > 0:
88  args = []
89  for arg in argv:
90  if ' ' in arg: arg = '"' + arg + '"'
91  args.append(arg)
92  print(' '.join(args))
93  return execfunc(argv)
94 
95 # ----------------------------------------------------------------------------
96 def call(argv, verbose=0):
97  """Execute MIRTK command and return exit code."""
98  return _call(argv, verbose=verbose, execfunc=subprocess.call)

100 # ----------------------------------------------------------------------------
101 def check_call(argv, verbose=0):
102  """Execute MIRTK command and throw exception on error."""
103  return _call(argv, verbose=verbose, execfunc=subprocess.check_call)

105 # ----------------------------------------------------------------------------
