81 /// Convert string to vtkDataSetAttributes::AttributeType
82 template <>
83 inline bool FromString(const char *str, vtkDataSetAttributes::AttributeTypes &type)
84 {
85  string name = ToLower(str);
86  if (name == "scalars") type = vtkDataSetAttributes::SCALARS;
87  else if (name == "vectors") type = vtkDataSetAttributes::VECTORS;
88  else if (name == "normals") type = vtkDataSetAttributes::NORMALS;
89  else if (name == "tcoords") type = vtkDataSetAttributes::TCOORDS;
90  else if (name == "tensors") type = vtkDataSetAttributes::TENSORS;
91  else if (name == "globalids") type = vtkDataSetAttributes::GLOBALIDS;
92  else if (name == "pedigreeids") type = vtkDataSetAttributes::PEDIGREEIDS;
93  else if (name == "edgeflag") type = vtkDataSetAttributes::EDGEFLAG;
94  else type = vtkDataSetAttributes::NUM_ATTRIBUTES;
95  return (type < vtkDataSetAttributes::NUM_ATTRIBUTES);
96 }

98 // -----------------------------------------------------------------------------
99 /// Convert vtkDataSetAttributes::AttributeType to string
100 template <>
101 inline string ToString(const vtkDataSetAttributes::AttributeTypes &type,
102  int w, char c, bool left)
103 {
104  const char *str;
105  switch (type) {
106  case vtkDataSetAttributes::SCALARS: str = "SCALARS"; break;
107  case vtkDataSetAttributes::VECTORS: str = "VECTORS"; break;
108  case vtkDataSetAttributes::NORMALS: str = "NORMALS"; break;
109  case vtkDataSetAttributes::TCOORDS: str = "TCOORDS"; break;
110  case vtkDataSetAttributes::TENSORS: str = "TENSORS"; break;
111  case vtkDataSetAttributes::GLOBALIDS: str = "GLOBALIDS"; break;
112  case vtkDataSetAttributes::PEDIGREEIDS: str = "PEDIGREEIDS"; break;
113  case vtkDataSetAttributes::EDGEFLAG: str = "EDGEFLAG"; break;
114  default: str = "UNKNOWN"; break;
115  }
116  return ToString(str, w, c, left);
117 }
