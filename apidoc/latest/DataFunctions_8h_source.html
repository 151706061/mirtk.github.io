185 /// Base class of element-wise data transformations
186 class ElementWiseBinaryOp : public Op
187 {
188  /// Constant value to add
189  mirtkPublicAttributeMacro(double, Constant);
190 
191  /// File path of second dataset
192  mirtkPublicAttributeMacro(string, FileName);
193 
194 private:
195 
196  double *_Data;
197  bool *_Mask;
198  double *_Other;
199 
200 protected:
201 
202  /// Constructor
203  ElementWiseBinaryOp(double value) : _Constant(value) {}
204 
205  /// Constructor
206  ElementWiseBinaryOp(const char *fname) : _Constant(.0), _FileName(fname) {}
207 
208 public:
209 
210  // Called by TBB's parallel_for, for internal use only!
211  void operator()(const blocked_range<int> &re) const
212  {
213  double *data = _Data + re.begin();
214  if (_Other) {
215  double *other = _Other + re.begin();
216  if (_Mask) {
217  bool *mask = _Mask + re.begin();
218  for (int i = re.begin(); i != re.end(); ++i) {
219  if (*mask) *data = this->Op(*data, *other, *mask);
220  ++data, ++other, ++mask;
221  }
222  } else {
223  bool mask = true;
224  for (int i = re.begin(); i != re.end(); ++i) {
225  *data = this->Op(*data, *other, mask);
226  ++data, ++other;
227  }
228  }
229  } else {
230  if (_Mask) {
231  bool *mask = _Mask + re.begin();
232  for (int i = re.begin(); i != re.end(); ++i) {
233  if (*mask) *data = this->Op(*data, _Constant, *mask);
234  ++data, ++mask;
235  }
236  } else {
237  bool mask = true;
238  for (int i = re.begin(); i != re.end(); ++i) {
239  *data = this->Op(*data, _Constant, mask);
240  ++data;
241  }
242  }
243  }
244  }
245 
246  /// Transform data value and/or mask it by setting mask = false
247  virtual double Op(double value, double, bool &) const = 0;
248 
249  /// Process given data (not thread-safe!)
250  virtual void Process(int n, double *data, bool *mask = NULL)
251  {
252  _Data = data;
253  _Mask = mask;
254  _Other = NULL;
255  if (!_FileName.empty()) {
256  if (Read(_FileName.c_str(), _Other) != n) {
257  cerr << "Input file " << _FileName << " has different number of data points!" << endl;
258  exit(1);
259  }
260  }
261  // MUST be called in the base class which defines Op!
262  //parallel_for(blocked_range<int>(0, n), *this);
263  }
