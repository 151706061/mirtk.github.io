131  /// Test stopping criteria
132  ///
133  /// The objective function must be up-to-date when this function is called.
134  /// This is usually the case anyway because the current objective function
135  /// value after a change of the parameters must be evaluated before this
136  /// function is called to be able to provide this value as argument.
137  ///
138  /// \note The objective function value may be NaN in case of a non-parametric
139  /// deformable surface model. In this case, stopping criteria are based
140  /// only on the current surface geometry or last node displacements.
141  /// Stopping criteria based on the objective function value should
142  /// never be fulfilled in this case and always return \c false.
143  ///
144  /// \param[in] iter Current number of iterations.
145  /// \param[in] prev Objective function value at previous iteration.
146  /// \param[in] value Objective function value at current iteration.
147  /// \param[in] delta Last change of objective function parameters.
148  ///
149  /// \returns True when at least one stopping criterion is fulfilled.
150  virtual bool Converged(int iter, double prev, double value, const double *delta);

196 // -----------------------------------------------------------------------------
197 inline bool LocalOptimizer::IsImprovement(double prev, double value) const
198 {
199  if (IsInf(prev)) return !IsNaN(value);
200  if (IsInf(value)) return false;
201  return prev - value > _Epsilon;
202 }
