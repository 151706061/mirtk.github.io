29 // ----------------------------------------------------------------------------
30 /// Image extrapolation modes
31 enum ExtrapolationMode {
32  Extrapolation_Default, // i.e., not specified
33  Extrapolation_None, // i.e., partial interpolation or default value
34  Extrapolation_Const,
35  Extrapolation_NN,
36  Extrapolation_Repeat, // i.e., periodic
37  Extrapolation_Mirror,
38  Extrapolation_ConstWithPeriodicTime,
39  // Add new enumeration values above
40  Extrapolation_Last
41 };
42 
43 // ----------------------------------------------------------------------------
44 inline string ToString(const ExtrapolationMode &m)
45 {
46  switch(m) {
47  case Extrapolation_Default: return "Default";
48  case Extrapolation_None: return "None";
49  case Extrapolation_Const: return "Const";
50  case Extrapolation_NN: return "NN";
51  case Extrapolation_Repeat: return "Repeat";
52  case Extrapolation_Mirror: return "Mirror";
53  case Extrapolation_ConstWithPeriodicTime: return "ConstWithPeriodicTime";
54  default: return "Unknown";
55  }
56 }
57 
58 // ----------------------------------------------------------------------------
59 /// Get corresponding extrapolation with periodic time
60 inline ExtrapolationMode ExtrapolationWithPeriodicTime(ExtrapolationMode m)
61 {
62  switch(m) {
63  case Extrapolation_ConstWithPeriodicTime:
64  case Extrapolation_Const: return Extrapolation_ConstWithPeriodicTime;
65  case Extrapolation_Repeat: return Extrapolation_Repeat;
66  default: return Extrapolation_None;
67  }
68 }
69 
70 // ----------------------------------------------------------------------------
71 /// Get corresponding extrapolation without periodic time
72 inline ExtrapolationMode ExtrapolationWithoutPeriodicTime(ExtrapolationMode m)
73 {
74  switch(m) {
75  case Extrapolation_Const:
76  case Extrapolation_ConstWithPeriodicTime: return Extrapolation_Const;
77  // Note: Extrapolation_Repeat remains periodic in all dimensions
78  default: return m;
79  }
80 }
81 
82 // ----------------------------------------------------------------------------
83 inline bool FromString(const char *str, ExtrapolationMode &m)
84 {
85  if (strcmp(str, "Default") == 0) m = Extrapolation_Default;
86  else if (strcmp(str, "None") == 0) m = Extrapolation_None;
87  else if (strcmp(str, "Const") == 0) m = Extrapolation_Const;
88  else if (strcmp(str, "NN") == 0) m = Extrapolation_NN;
89  else if (strcmp(str, "Repeat") == 0) m = Extrapolation_Repeat;
90  else if (strcmp(str, "Mirror") == 0) m = Extrapolation_Mirror;
91  else if (strcmp(str, "ConstWithPeriodicTime") == 0) m = Extrapolation_ConstWithPeriodicTime;
92  else return false;
93  return true;
94 }
