31 /**
32  * Class for Gaussian blurring of images
33  *
34  * This class defines and implements the Gaussian blurring of images. The
35  * blurring is implemented by three successive 1D convolutions with a 1D
36  * Gaussian kernel.
37  *
38  * By default, if one isotropic Gaussian standard deviation is specified,
39  * the first three dimensions of the image are blurred only. Otherwise,
40  * the 1D convolution with a 1D Gaussian kernel is performed only for
41  * dimensions of more than one voxel size and for which a non-zero
42  * standard deviation for the Gaussian kernel has been set.
43  */
44 template <class VoxelType>
45 class GaussianBlurring : public ImageToImage<VoxelType>
46 {
47  mirtkInPlaceImageFilterMacro(GaussianBlurring);
48 
49  /// Standard deviation of Gaussian kernel in x
50  mirtkAttributeMacro(double, SigmaX);
51 
52  /// Standard deviation of Gaussian kernel in y
53  mirtkAttributeMacro(double, SigmaY);
54 
55  /// Standard deviation of Gaussian kernel in z
56  mirtkAttributeMacro(double, SigmaZ);
57 
58  /// Standard deviation of Gaussian kernel in t
59  mirtkAttributeMacro(double, SigmaT);
60 
61 protected:
62 
63  /// Gaussian convolution kernel
64  GenericImage<RealPixel> *_Kernel;
65 
66  /// Initialize filter
67  virtual void Initialize();
68 
69  /// Initialize 1D Gaussian kernel with sigma given in voxel units
70  virtual void InitializeKernel(double);
71 
72  /// Finalize filter
73  virtual void Finalize();
74 
75 public:
76 
77  /// Constructor
78  GaussianBlurring(double = 1.0);
79 
80  /// Constructor
81  GaussianBlurring(double, double, double = .0, double = .0);
82 
83  /// Destructor
84  ~GaussianBlurring();

86  /// Run Gaussian blurring
87  virtual void Run();
88

97

99  virtual void RunT();
100

103 
104  /// Set sigma

106

108  static int KernelSize(double);
109
