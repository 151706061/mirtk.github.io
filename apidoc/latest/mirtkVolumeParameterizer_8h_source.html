36 /**
37  * Base class of filters which parameterize the interior of a
38  * piecewise linear complex (PLC)
39  */
40 class VolumeParameterizer : public Object
41 {
42  mirtkAbstractMacro(VolumeParameterizer);
43 
44  // ---------------------------------------------------------------------------
45  // Attributes
46 
47  /// Input point set (e.g., surface mesh or tetrahedral mesh)
48  mirtkPublicAttributeMacro(vtkSmartPointer<vtkPointSet>, InputSet);
49 
50  /// Input boundary map
51  mirtkPublicAttributeMacro(vtkSmartPointer<vtkDataArray>, InputMap);
52 
53  /// Boundary surface of input point set
54  mirtkAttributeMacro(vtkSmartPointer<vtkPolyData>, Boundary);
55 
56  /// Boundary surface map
57  mirtkAttributeMacro(vtkSmartPointer<vtkDataArray>, BoundaryMap);
58 
59  /// Volumetric output map
60  /// \note The output map is uninitialized!
61  mirtkReadOnlyComponentMacro(VolumetricMap, OutputMap);
62 
63  /// Copy attributes of this class from another instance
64  void CopyAttributes(const VolumeParameterizer &);
65 
66  // ---------------------------------------------------------------------------
67  // Construction/Destruction
68 
69 protected:
70 
71  /// Default constructor
72  VolumeParameterizer();
73 
74  /// Copy constructor
75  VolumeParameterizer(const VolumeParameterizer &);
76 
77  /// Assignment operator
78  VolumeParameterizer &operator =(const VolumeParameterizer &);
79 
80 public:
81 
82  /// Destructor
83  virtual ~VolumeParameterizer();
84 
85  /// Dimension of output domain of volumetric map
86  int OutputDimension() const;
87 
88  /// Get volumetric output map and set _OutputMap to NULL
89  ///
90  /// \note The returned object has to be deleted by the caller.
91  /// Use OutputMap() instead to keep ownership with this class.
92  VolumetricMap *GetOutputMap();
93 
94  // ---------------------------------------------------------------------------
95  // Execution
96 
97  /// Parameterize interior of input data set
98  void Run();
99 
100 protected:

103  virtual void Initialize();
104 
105  /// Initialize boundary surface with corresponding boundary map as point data
106  virtual void InitializeBoundary(vtkPointSet *, vtkDataArray *);
107 
108  /// Parameterize interior of input data set
109  virtual void Parameterize() = 0;
110 
111  /// Finalize filter execution
112  virtual void Finalize();
113 
114 };
