30 enum InternalForceTerm
31 {
32  IFT_Unknown = IFT_Begin, ///< Unknown/invalid internal force
33  IFT_Distortion = EM_MetricDistortion, ///< Minimize metric distortion
34  IFT_Stretching = EM_Stretching, ///< Stretching force (rest edge length)
35  IFT_Curvature = EM_Curvature, ///< Minimize curvature of point set surface
36  IFT_NonSelfIntersection = EM_NonSelfIntersection, ///< Repels too close non-neighboring triangles
37  IFT_Repulsion = EM_RepulsiveForce, ///< Repels too close non-neighboring nodes
38  IFT_Inflation = EM_InflationForce, ///< Inflate point set surface
39 };
40 
41 // -----------------------------------------------------------------------------
42 inline string ToString(const InternalForceTerm &ift)
43 {
44  EnergyMeasure em = static_cast<EnergyMeasure>(ift);
45  if (em <= IFT_Begin || em >= IFT_End) return "Unknown";
46  return ToString(em);
47 }
48 
49 // -----------------------------------------------------------------------------
50 inline bool FromString(const char *str, InternalForceTerm &ift)
51 {
52  EnergyMeasure em = EM_Unknown;
53  if (FromString(str, em) && IFT_Begin < em && em < IFT_End) {
54  ift = static_cast<InternalForceTerm>(em);
55  return true;
56  } else {
57  ift = IFT_Unknown;
58  return false;
59  }
60 }
