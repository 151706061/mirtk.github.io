97 /// NIfTI-1 xform codes to describe the "standard" coordinate system

101  NIFTI_XFORM_SCANNER_ANAT = 1, ///< Scanner-based anatomical coordinates

128 /**
129  * NIfTI-1 image header information.
130  *
131  * \note Only contains most commonly used entries of nifti_image struct
132  * and uses MIRTK types for non-builtin types.
133  */
134 struct NiftiImageInfo
135 {
136  int ndim; ///< last dimension greater than 1 (1..7)
137  int nx; ///< dimensions of grid array
138  int ny; ///< dimensions of grid array
139  int nz; ///< dimensions of grid array
140  int nt; ///< dimensions of grid array
141  int nu; ///< dimensions of grid array
142  int nv; ///< dimensions of grid array
143  int nw; ///< dimensions of grid array
144  size_t nvox; ///< number of voxels = nx*ny*nz*...*nw
145  int nbyper; ///< bytes per voxel, matches datatype
146  int datatype; ///< type of data in voxels: DT_* code
147  float dx; ///< grid spacings
148  float dy; ///< grid spacings
149  float dz; ///< grid spacings
150  float dt; ///< grid spacings
151  float du; ///< grid spacings
152  float dv; ///< grid spacings
153  float dw; ///< grid spacings
154  float scl_slope; ///< scaling parameter - slope
155  float scl_inter; ///< scaling parameter - intercept
156  float cal_min; ///< calibration parameter, minimum
157  float cal_max; ///< calibration parameter, maximum
158  int slice_code; ///< code for slice timing pattern
159  int slice_start; ///< index for start of slices
160  int slice_end; ///< index for end of slices
161  float slice_duration; ///< time between individual slices
162  int qform_code; ///< codes for (x,y,z) space meaning
163  Matrix qto_xyz; ///< qform: transform (i,j,k) to (x,y,z)
164  Matrix qto_ijk; ///< qform: transform (x,y,z) to (i,j,k)
165  int sform_code; ///< codes for (x,y,z) space meaning
166  Matrix sto_xyz; ///< sform: transform (i,j,k) to (x,y,z)
167  Matrix sto_ijk; ///< sform: transform (x,y,z) to (i,j,k)
168  float toffset; ///< time coordinate offset
169  int xyz_units; ///< dx,dy,dz units: NIFTI_UNITS_* code
170  int time_units; ///< dt units: NIFTI_UNITS_* code
171  int nifti_type; ///< 0==ANALYZE, 1==NIFTI-1 (1 file),
172  /// 2==NIFTI-1 (2 files),
173  /// 3==NIFTI-ASCII (1 file)
174  int intent_code; ///< statistic type (or something)
175  float intent_p1; ///< intent parameters
176  float intent_p2; ///< intent parameters
177  float intent_p3; ///< intent parameters
178  string intent_name; ///< optional description of intent data
179  string descrip; ///< optional text to describe dataset
180  string aux_file; ///< auxiliary filename
181  string fname; ///< header filename (.hdr or .nii)
182  string iname; ///< image filename (.img or .nii)
183  int iname_offset; ///< offset into iname where data starts
184  int swapsize; ///< swap unit in image data (might be 0)
185  int byteorder; ///< byte order on disk (MSB_ or LSB_FIRST)
186 
187  /// Constructor, reads header from NIfTI image file
188  NiftiImageInfo(const char * = nullptr);
189 };
