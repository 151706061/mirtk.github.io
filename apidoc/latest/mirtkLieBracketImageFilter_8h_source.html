98 template <class VoxelType>
99 void LieBracketImageFilter<VoxelType>::Input(int i, const GenericImage<VoxelType> *image)
100 {
101  if (i == 0) ImageToImage<VoxelType>::Input(image);
102  else if (i == 1) _Input2 = image;
103  else {
104  cerr << this->NameOfClass() << "::Input: Input index out of range: " << i << endl;
105  exit(1);
106  }
107 }

109 // --------------------------------------------------------------------------
110 template <class VoxelType>
111 const GenericImage<VoxelType> *LieBracketImageFilter<VoxelType>::Input(int i) const
112 {
113  if (i == 0) return ImageToImage<VoxelType>::Input();
114  else if (i == 1) return _Input2;
115  else {
116  cerr << this->NameOfClass() << "::Input: Input index out of range: " << i << endl;
117  exit(1);
118  }
119 }

128 // --------------------------------------------------------------------------
129 template <class VoxelType>
130 void LieBracketImageFilter<VoxelType>::Initialize()
131 {
132  // Initialize base class
133  ImageToImage<VoxelType>::Initialize();
134 
135  // Check second input
136  if (_Input2 == NULL) {
137  cerr << this->NameOfClass() << "::Initialize: Filter has no second input" << endl;
138  exit(1);
139  }
140  if (!_Input2->HasSpatialAttributesOf(this->_Input) || _Input2->T() != this->_Input->T()) {
141  cerr << this->NameOfClass() << "::Initialize: Attributes of input images do not match" << endl;
142  exit(1);
143  }
144 }

169 template <class VoxelType>
170 LieBracketImageFilter<VoxelType> *
171 LieBracketImageFilter<VoxelType>::New(const ImageAttributes &attr, bool usejac)
172 {
173  if (attr._z > 1) {
174  if (usejac) return new LieBracketImageFilter3D<VoxelType>;
175  else return new DifferenceOfCompositionLieBracketImageFilter3D<VoxelType>;
176  } else {
177  if (usejac) return new LieBracketImageFilter2D<VoxelType>;
178  else {
179  cerr << NameOfType() << "::New: DifferenceOfCompositionLieBracketImageFilter2D not implemented" << endl;
180  exit(1);
181  }
182  }
183 }

193 // ----------------------------------------------------------------------------
194 template <class VoxelType>
195 void liebracket(GenericImage<VoxelType> *ov,
196  GenericImage<VoxelType> *lv,
197  GenericImage<VoxelType> *rv, bool usejac = true)
198 {
199  typedef LieBracketImageFilter<VoxelType> LieBracketFilter;
200  unique_ptr<LieBracketFilter> filter(LieBracketFilter::New(ov, usejac));
201  filter->Input(0, lv);
202  filter->Input(1, rv);
203  filter->Output(ov);
204  filter->Run();
205 }
