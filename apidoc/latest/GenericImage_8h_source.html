488 // -----------------------------------------------------------------------------
489 template <class VoxelType> template <class VoxelType2>
490 GenericImage<VoxelType>& GenericImage<VoxelType>::operator=(const GenericImage<VoxelType2> &image)
491 {
492  this->Initialize(image.GetImageAttributes());
493  VoxelType *ptr1 = this->GetPointerToVoxels();
494  const VoxelType2 *ptr2 = image.GetPointerToVoxels();
495  for (int idx = 0; idx < _NumberOfVoxels; idx++) {
496  ptr1[idx] = voxel_cast<VoxelType>(ptr2[idx]);
497  }
498  if (image.OwnsMask()) {
499  _mask = new BinaryImage(*image.GetMask());
500  _maskOwner = true;
501  } else {
502  _mask = const_cast<BinaryImage *>(image.GetMask());
503  _maskOwner = false;
504  }
505  if (image.HasBackgroundValue()) {
506  this->PutBackgroundValueAsDouble(image.GetBackgroundValueAsDouble());
507  }
508  return *this;
509 }

614 // -----------------------------------------------------------------------------
615 template <class VoxelType> template <class VoxelType2>
616 bool GenericImage<VoxelType>::operator==(const GenericImage<VoxelType2> &image) const
617 {
618  if (this->GetImageAttributes() != image.GetImageAttributes()) return false;
619  const VoxelType *ptr1 = this->GetPointerToVoxels();
620  const VoxelType2 *ptr2 = image.GetPointerToVoxels();
621  for (int idx = 0; idx < image; ++idx) {
622  if (IsForeground(idx) && image.IsForeground(idx) && ptr1[idx] != ptr2[idx]) {
623  return false;
624  }
625  }
626  return true;
627 }
