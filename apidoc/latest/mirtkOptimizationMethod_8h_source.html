30 /// Enumeration of available optimization methods
31 enum OptimizationMethod
32 {
33  OM_DownhillDescent,
34  OM_GradientDescent,
35  OM_GradientDescentConstrained,
36  OM_SteepestGradientDescent,
37  OM_ConjugateGradientDescent,
38  OM_ClosedForm,
39  OM_LBFGS,
40  OM_LineSearch,
41  OM_EulerMethod, ///< Explicit Euler method for deformable surface models
42  OM_EulerMethodWithDamping, ///< Explicit Euler method with momentum for deformable surface models
43  OM_EulerMethodWithMomentum ///< Explicit Euler method with momentum for deformable surface models
44 };
45 
46 // -----------------------------------------------------------------------------
47 inline string ToString(const OptimizationMethod &m)
48 {
49  switch (m) {
50  case OM_DownhillDescent: return "DownhillDescent";
51  case OM_GradientDescent: return "GradientDescent";
52  case OM_GradientDescentConstrained: return "GradientDescentConstrained";
53  case OM_SteepestGradientDescent: return "SteepestGradientDescent";
54  case OM_ConjugateGradientDescent: return "ConjugateGradientDescent";
55  case OM_ClosedForm: return "ClosedForm";
56  case OM_LBFGS: return "LBFGS";
57  case OM_LineSearch: return "LineSearch";
58  case OM_EulerMethod: return "EulerMethod";
59  case OM_EulerMethodWithDamping: return "EulerMethodWithDamping";
60  case OM_EulerMethodWithMomentum: return "EulerMethodWithMomentum";
61  default: return "Unknown";
62  }
63 }
64 
65 // -----------------------------------------------------------------------------
66 inline bool FromString(const char *str, OptimizationMethod &m)
67 {
68  if (strcmp(str, "DownhillDescent") == 0 ||
69  strcmp(str, "Downhill") == 0) {
70  m = OM_DownhillDescent;
71  } else if (strcmp(str, "GradientDescent") == 0 ||
72  strcmp(str, "GD") == 0) {
73  m = OM_GradientDescent;
74  } else if (strcmp(str, "ConstrainedGradientDescent") == 0 ||
75  strcmp(str, "GradientDescentConstrained") == 0) {
76  m = OM_GradientDescentConstrained;
77  } else if (strcmp(str, "SteepestGradientDescent") == 0 ||
78  strcmp(str, "SteepestGradient") == 0 ||
79  strcmp(str, "SGD") == 0) {
80  m = OM_SteepestGradientDescent;
81  } else if (strcmp(str, "ConjugateGradientDescent") == 0 ||
82  strcmp(str, "ConjugateGradient") == 0 ||
83  strcmp(str, "CGD") == 0) {
84  m = OM_ConjugateGradientDescent;
85  } else if (strcmp(str, "ClosedForm") == 0) {
86  m = OM_ClosedForm;
87  } else if (strcmp(str, "LBFGS") == 0) {
88  m = OM_LBFGS;
89  } else if (strcmp(str, "LineSearch") == 0) {
90  m = OM_LineSearch;
91  } else if (strcmp(str, "EulerMethod") == 0) {
92  m = OM_EulerMethod;
93  } else if (strcmp(str, "EulerMethodWithDamping") == 0) {
94  m = OM_EulerMethodWithDamping;
95  } else if (strcmp(str, "EulerMethodWithMomentum") == 0) {
96  m = OM_EulerMethodWithMomentum;
97  } else {
98  return false;
99  }
100  return true;
101 }
