43 /**
44  * Base class for homogeneous transformations.
45  *
46  * This class defines and implements homogeneous transformations which can
47  * be represented by a 4 x 4 transformation matrix. The transformation of
48  * a point is implemented by post multiplying the transformation matrix with
49  * the point in homogeneous coordinates. The transformation is parameterized
50  * by twelve degrees of freedom.
51  *
52  */

273 inline void HomogeneousTransformation::Update(AttributeSelector what)
274 {
275  // Update attributes
276  switch (what) {
277  case MATRIX:
278  this->UpdateMatrix();
279  break;
280  case DOFS:
281  this->UpdateDOFs();
282  break;
283  case DOFS_MATRIX:
284  this->UpdateDOFs();
285  this->UpdateMatrix();
286  break;
287  }
288  // Mark transformation as changed
289  this->Changed(true);
290  // Notify observers, keeping changed state intact
291  Broadcast(ModifiedEvent, &_matrix);
292 }

308 // -----------------------------------------------------------------------------
309 inline void HomogeneousTransformation::Add(const DOFValue *dx)
310 {
311  double delta, max_delta = .0;
312  for (int idx = 0; idx < _NumberOfDOFs; ++idx) {
313  // Add change also to passive parameters
314  _Param[idx] += dx[idx];
315  delta = abs(dx[idx]);
316  if (delta > max_delta) max_delta = delta;
317  }
318  if (max_delta > .0) this->Update(MATRIX);
319 }
320 
321 // -----------------------------------------------------------------------------
322 inline double HomogeneousTransformation::Update(const DOFValue *dx)
323 {
324  double delta, max_delta = .0;
325  for (int idx = 0; idx < _NumberOfDOFs; ++idx) {
326  if (_Status[idx] == Active) {
327  _Param[idx] += dx[idx];
328  delta = abs(dx[idx]);
329  if (delta > max_delta) max_delta = delta;
330  }
331  }
332  if (max_delta > .0) this->Update(MATRIX);
333  return max_delta;
334 }

341  this->Update(DOFS_MATRIX); // also updates _inverse
342 }

361 inline void HomogeneousTransformation::GlobalTransform(double &x, double &y, double &z, double, double) const
362 {
363  double a = _matrix(0, 0) * x + _matrix(0, 1) * y + _matrix(0, 2) * z + _matrix(0, 3);
364  double b = _matrix(1, 0) * x + _matrix(1, 1) * y + _matrix(1, 2) * z + _matrix(1, 3);
365  double c = _matrix(2, 0) * x + _matrix(2, 1) * y + _matrix(2, 2) * z + _matrix(2, 3);
366 
367  x = a;
368  y = b;
369  z = c;
370 }

385 inline void HomogeneousTransformation::GlobalInverse(double &x, double &y, double &z, double, double) const
386 {
387  double a = _inverse(0, 0) * x + _inverse(0, 1) * y + _inverse(0, 2) * z + _inverse(0, 3);
388  double b = _inverse(1, 0) * x + _inverse(1, 1) * y + _inverse(1, 2) * z + _inverse(1, 3);
389  double c = _inverse(2, 0) * x + _inverse(2, 1) * y + _inverse(2, 2) * z + _inverse(2, 3);
390 
391  x = a;
392  y = b;
393  z = c;
394 }
