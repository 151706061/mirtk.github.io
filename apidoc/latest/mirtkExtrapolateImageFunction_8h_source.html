33 /**
34  * Abstract base class for discrete image extrapolation
35  *
36  * This abstract base class defines a common interface for image functions
37  * which provide a value at each discrete image location, especially also
38  * those which are outside the finite domain on which the image is defined.
39  * At these outside voxels, the image function extrapolates the image values,
40  * thus extending an image function to an infinite discrete lattice. An image
41  * extrapolate image function is mainly used indirectly through an instance of
42  * InterpolateImageFunction. Image values at continuous voxel positions
43  * are interpolated by the interpolate image function using the extrapolated
44  * image values at discrete lattice points.
45  */
46 class ExtrapolateImageFunction : public ImageFunction
47 {
48  mirtkAbstractMacro(ExtrapolateImageFunction);

65  const BaseImage * = NULL);

130 /**
131  * Generic base class for image extrapolation functions
132  *
133  * This base class is templated over the type of the input image to be extrapolated
134  * and thus subclasses can make use of image voxel type specific (non-virtual)
135  * getters to access the image data. No conversion of the voxel type to a uniform
136  * voxel type such as double scalar or vector type is required when the type of
137  * the image to be extrapolated is known. Otherwise, if the image type is unknown,
138  * use the abstract ExtrapolateImageFunction interface instead.
139  *
140  * \sa ExtrapolateImageFunction
141  */
142 template <class TImage>
143 class GenericExtrapolateImageFunction : public ExtrapolateImageFunction
144 {
145  mirtkAbstractMacro(GenericExtrapolateImageFunction);
146 
147  // ---------------------------------------------------------------------------

152  typedef TImage ImageType; ///< Input image type

154  typedef typename ImageType::RealType RealType; ///< Compatible floating-point type
155 
156  // ---------------------------------------------------------------------------
157  // Construction/Destruction

159 protected:

173  // ---------------------------------------------------------------------------

176  /// Set input image

463 // -----------------------------------------------------------------------------
464 inline bool ExtrapolateImageFunction::IsForeground(int i, int j, int k, int l) const
465 {
466  if (!this->Input()->HasBackgroundValue()) return true;
467  const double value = this->GetAsDouble(i, j, k, l);
468  const double bg = this->Input()->GetBackgroundValueAsDouble();
469  return (value != bg && (!IsNaN(value) || !IsNaN(bg)));
470 }

731 // -----------------------------------------------------------------------------
732 template <class TImage>
733 GenericExtrapolateImageFunction<TImage> *
734 GenericExtrapolateImageFunction<TImage>
735 ::New(enum ExtrapolationMode mode, const TImage *image)
736 {
737  GenericExtrapolateImageFunction<TImage> *p = NULL;
738  switch (mode) {
739  case Extrapolation_None: { p = NULL; break; }
740  case Extrapolation_Const: { p = new GenericConstExtrapolateImageFunction <TImage>(); break; }
741  case Extrapolation_NN: { p = new GenericNearestNeighborExtrapolateImageFunction<TImage>(); break; }
742  case Extrapolation_Repeat: { p = new GenericRepeatExtrapolateImageFunction <TImage>(); break; }
743  case Extrapolation_Mirror: { p = new GenericMirrorExtrapolateImageFunction <TImage>(); break; }
744  case Extrapolation_ConstWithPeriodicTime:
745  p = new GenericConstExtrapolateImageFunctionWithPeriodicTime<TImage>();
746  break;
747  default:
748  cerr << "GenericExtrapolateImageFunction::New: Unknown extrapolation mode: " << mode << endl;
749  exit(1);
750  }
751  if (p) p->Input(image);
752  return p;
753 }
